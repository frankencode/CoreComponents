48 void Crc16::feed(const void *buf, int bufFill)
49 {
50  for (int i = 0; i < bufFill; ++i)
51  crc_ = crcTable[(crc_ ^ reinterpret_cast<const uint8_t*>(buf)[i]) & 0xFF] ^ (crc_ >> 8);
52 }
