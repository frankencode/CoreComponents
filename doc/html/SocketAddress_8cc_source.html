27 Ref<SocketAddress> SocketAddress::read(String location)
28 {
29  Ref<SocketAddress> address;
30 
31 
32  Ref<Uri> uri;
33 
34  try {
35  uri = Uri::parse(location);
36  }
37  catch (UriSyntaxError &ex) {
38  throw LocationSyntaxError(Format("Invalid address syntax: \"%%\"") << ex.text());
39  }
40 
41  if (uri->hostIsNumeric()) {
42  address = SocketAddress::create(uri->family(), uri->host(), uri->port());
43  }
44  else {
45  Ref<SocketAddressList> addressList = SocketAddress::resolve(uri->host());
46  if (addressList->count() > 0) {
47  address = addressList->at(0);
48  address->setPort(uri->port());

50  else throw HostNameResolutionError(Format("Failed to resolve host name \"%%\"") << uri->host());
51  }
52 
53  return address;
54 }

61 SocketAddress::SocketAddress(int family, String address, int port):
62  socketType_(0),
63  protocol_(0)
64 {
65  void *addr = 0;
66 
67  if (family == AF_INET) {
68  // inet4Address_.sin_len = sizeof(addr);
69  *(uint8_t *)&inet4Address_ = sizeof(inet4Address_); // uggly, but safe HACK, for BSD4.4
70  inet4Address_.sin_family = AF_INET;
71  inet4Address_.sin_port = htons(port);
72  inet4Address_.sin_addr.s_addr = htonl(INADDR_ANY);
73  addr = &inet4Address_.sin_addr;
74  }
75  else if (family == AF_INET6) {
76  #ifdef SIN6_LEN
77  inet6Address_.sin6_len = sizeof(inet6Address_);
78  #endif
79  inet6Address_.sin6_family = AF_INET6;
80  inet6Address_.sin6_port = htons(port);
81  inet6Address_.sin6_addr = in6addr_any;
82  addr = &inet6Address_.sin6_addr;
83  }
84  else if (family == AF_LOCAL) {
85  localAddress_.sun_family = AF_LOCAL;
86  if (unsigned(address->count()) + 1 > sizeof(localAddress_.sun_path))
87  CC_DEBUG_ERROR("Socket path exceeds maximum length");
88  if (address == "" || address == "*" || address == "::")
89  localAddress_.sun_path[0] = 0;
90  else
91  memcpy(localAddress_.sun_path, address->chars(), address->count() + 1);
92  }
93  else if (family != AF_UNSPEC)
94  CC_DEBUG_ERROR("Unsupported address family");
95 
96  if (family != AF_LOCAL) {
97  if (address != "" && address != "*" && address != "::") {
98  if (family == AF_UNSPEC) {
99  if (address->count(':') >= 2) family = AF_INET6;
100  else family = AF_INET;
101  }
102  if (inet_pton(family, address, addr) != 1)
103  CC_DEBUG_ERROR("Invalid address string");
104  }
105  }
106 }

164 String SocketAddress::networkAddress() const
165 {
166  String s;
167  if (addr_.sa_family == AF_LOCAL) {
168  s = localAddress_.sun_path;
169  }
170  else {
171  const int bufSize = INET_ADDRSTRLEN + INET6_ADDRSTRLEN;
172  char buf[bufSize];
173 
174  const void *addr = 0;
175  const char *sz = 0;
176 
177  if (addr_.sa_family == AF_INET)
178  addr = &inet4Address_.sin_addr;
179  else if (addr_.sa_family == AF_INET6)
180  addr = &inet6Address_.sin6_addr;
181  else
182  CC_DEBUG_ERROR("Unsupported address family");
183 
184  sz = inet_ntop(addr_.sa_family, const_cast<void *>(addr), buf, bufSize);
185  if (!sz)
186  CC_DEBUG_ERROR("Illegal binary address format");
187 
188  s = sz;
189  }

210 Ref<SocketAddressList> SocketAddress::resolve(String hostName, String serviceName, int family, int socketType, String *canonicalName)
211 {
212  addrinfo hint;
213  addrinfo *head = 0;
214 
215  memclr(&hint, sizeof(hint));
216  if ((hostName == "*") || (hostName == "")) hint.ai_flags |= AI_PASSIVE;
217  hint.ai_flags |= (canonicalName && (hint.ai_flags & AI_PASSIVE) == 0) ? AI_CANONNAME : 0;
218  hint.ai_family = (hostName == "*") ? AF_INET : family;
219  hint.ai_socktype = socketType;
220 
221  int ret;
222  {
223  char *n = 0;
224  char *s = 0;
225  if ((hint.ai_flags & AI_PASSIVE) == 0) n = hostName;
226  if (serviceName != "") s = serviceName;
227  ret = getaddrinfo(n, s, &hint, &head);
228  }
229 
230  if (ret != 0 && ret != EAI_NONAME)
231  CC_DEBUG_ERROR(gai_strerror(ret));
232 
233  Ref<SocketAddressList> list = SocketAddressList::create();
234 
235  if (canonicalName) {
236  if (head) {
237  if (head->ai_canonname)
238  *canonicalName = head->ai_canonname;
239  }
240  }
241 
242  addrinfo *next = head;
243 
244  while (next) {
245  if ((next->ai_family == AF_INET) || (next->ai_family == AF_INET6))
246  list->append(new SocketAddress(next));
247  next = next->ai_next;
248  }
249 
250  if (head)
251  freeaddrinfo(head);
252 
253  return list;
254 }

304 uint64_t SocketAddress::networkPrefix() const
305 {
306  uint64_t prefix = 0;
307  if (family() == AF_INET) {
308  prefix = endianGate(inet4Address_.sin_addr.s_addr);
309  }
310  else if (family() == AF_INET6) {
311  uint8_t const *a = inet6Address_.sin6_addr.s6_addr;
312  for (int i = 0; i < 8; ++i) {
313  prefix <<= 8;
314  prefix |= a[i];
315  }
316  }
317  return prefix;
318 }

320 bool SocketAddress::equals(const SocketAddress *b) const
321 {
322  if (family() != b->family()) return false;
323 
324  if (family() == AF_INET) {
325  return inet4Address_.sin_addr.s_addr == b->inet4Address_.sin_addr.s_addr;
326  }
327  else if (family() == AF_INET6) {
328  uint8_t const *x = inet6Address_.sin6_addr.s6_addr;
329  uint8_t const *y = b->inet6Address_.sin6_addr.s6_addr;
330  for (int i = 0; i < 8; ++i) {
331  if (x[i] != y[i]) return false;
332  }
333  }
334  else if (family() == AF_LOCAL) {
335  return strcmp(localAddress_.sun_path, b->localAddress_.sun_path) == 0;
336  }
337 
338  return false;
339 }
