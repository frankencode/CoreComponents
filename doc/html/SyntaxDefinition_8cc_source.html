43 Ref<StringList> SyntaxDefinition::split(const ByteArray *text) const
44 {
45  Ref<StringList> parts = StringList::create();
46  for (int i = 0; i < text->count();) {
47  Ref<SyntaxState> state = find(text, i);
48  if (state->valid()) {
49  parts->append(text->copy(i, state->i0()));
50  i = state->i1();
51  }
52  else {
53  parts->append(text->copy(i, text->count()));
54  break;
55  }
56  }
57  return parts;
58 }
