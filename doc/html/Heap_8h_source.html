15 template<class T, template<class> class Order = Ascending>
16 class GenericHeap: public Object, public Order<T>
17 {
18 public:

33  inline void push(const T &item)
34  {
35  CC_ASSERT(fill_ < size_);
36  buf_[fill_] = item;
37  ++fill_;
38  passUpLast();
39  }
40 
41  inline void pop(T *item)
42  {
43  CC_ASSERT(fill_ > 0);
44  *item = buf_[0];
45  --fill_;
46  buf_[0] = buf_[fill_];
47  passDownFromTop();
48  }

51  T item;
52  pop(&item);
53  return item;
54  }
55 
56  inline T top() { CC_ASSERT(!isEmpty()); return buf_[0]; }
57 
58  inline void clear() { fill_ = 0; }

60  inline static int parent(int i) { return (i - 1) / 2; }
61  inline static int leftChild(int i) { return 2 * i + 1; }
62  inline static int rightChild(int i) { return 2 * i + 2; }
63 
64  inline void xchg(int i, int j)
65  {
66  T h = buf_[i];
67  buf_[i] = buf_[j];
68  buf_[j] = h;
69  }
70 
71  inline int min(int i, int j, int k)
72  {
73  int h = Order<T>::below(buf_[i], buf_[j]) ? i : j;
74  return Order<T>::below(buf_[h], buf_[k]) ? h : k;
75  }
76 
77  void passUpLast()
78  {
79  if (fill_ == 1) return;
80  int i = fill_ - 1;
81  while (i != 0) {
82  int j = parent(i);
83  if (!Order<T>::below(buf_[i], buf_[j])) break;
84  xchg(i, j);
85  i = j;
86  }
87  }
88 
89  void passDownFromTop()
90  {
91  if (fill_ == 0) return;
92  int i = 0;
93  while (true) {
94  int j, lc, rc;
95  lc = leftChild(i);
96  rc = rightChild(i);
97  if (rc < fill_) {
98  j = Order<T>::below(buf_[lc], buf_[i]) ? lc : i;
99  j = Order<T>::below(buf_[rc], buf_[j]) ? rc : j;
100  if (j == i) break;
101  xchg(i, j);
102  i = j;
103  }
104  else if (lc < fill_) {
105  if (Order<T>::below(buf_[lc], buf_[i])) xchg(i, lc);
106  break;
107  }
108  else
109  break;
110  }
111  }
112 
113 protected:
114  GenericHeap(int size)
115  : fill_(0),
116  size_(size),
117  bufOwner_(true),
118  buf_(new T[size])
119  {}
