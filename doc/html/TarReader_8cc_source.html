36 bool TarReader::readHeader(Ref<ArchiveEntry> *nextEntry)
37 {
38  if (!data_) data_ = ByteArray::create(512);
39  *nextEntry = ArchiveEntry::create();
40 
41  ByteArray *data = data_;
42  ArchiveEntry *entry = *nextEntry;
43 
44  if (source_->readSpan(data->count(), data) < data->count()) return false;
45  i_ += data->count();
46 
47  bool eoi = true;
48  for (int i = 0; i < data->count(); ++i) {
49  if (data->byteAt(i) != 0) {
50  eoi = false;
51  break;
52  }
53  }
54 
55  if (eoi) return false;
56 
57  bool ustarMagic = false;
58  bool gnuMagic = false;
59 
60  bool readAgain = true;
61  while (readAgain) {
62  readAgain = false;
63 
64  String magic;
65  data->scanString(&magic, "", 257, 263);
66  ustarMagic = (magic == "ustar");
67  gnuMagic = (magic == "ustar ");
68 
69  unsigned checksum, probesum;
70  data->scan(&checksum, 8, 148, 156);
71  entry->type_ = data->at(156);
72  if (entry->path_ == "") data->scanString(&entry->path_, "", 0, 100);
73  if (entry->linkPath_ == "") data->scanString(&entry->linkPath_, "", 157, 257);
74 
75  probesum = tarHeaderSum(data);
76  if (checksum != probesum)
77  throw BrokenArchive(i_ - data->count(), Format("Checksum mismatch (%% != %%), path = \"%%\"") << oct(checksum, 6) << oct(probesum, 6) << entry->path());
78 
79  if (gnuMagic) {
80  while ((entry->type_ == 'K' || entry->type_ == 'L') /*&& entry->path_ == "././@LongLink"*/) {
81  data->scan(&entry->size_, 8, 124, 136);
82  String longPath = source_->readSpan(entry->size_);
83  if (longPath->count() < entry->size_)
84  throw BrokenArchive(i_, "Expected GNU @LongLink data");
85  i_ += entry->size_;
86  if (entry->size() % 512 != 0) {
87  i_ += source_->skip(512 - entry->size() % 512);
88  }
89  if (longPath->byteAt(longPath->count() - 1) == 0)
90  longPath->truncate(longPath->count() - 1);
91  if (entry->type_ == 'K') entry->linkPath_ = longPath;
92  else if (entry->type_ == 'L') entry->path_ = longPath;
93  if (source_->readSpan(data->count(), data) < data->count())
94  throw BrokenArchive(i_, "Expected GNU @LongLink header");
95  i_ += data->count();
96  entry->type_ = data->at(156);
97  readAgain = true;
98  }
99  }
100  }
101 
102  if (ustarMagic || gnuMagic) {
103  String prefix;
104  data->scanString(&entry->userName_, "", 265, 297);
105  data->scanString(&entry->groupName_, "", 297, 329);
106  if (!gnuMagic) {
107  data->scanString(&prefix, "", 345, 500);
108  if (prefix != "") entry->path_ = prefix + "/" + entry->path_;
109  }
110  }
111 
112  data->scan(&entry->mode_, 8, 100, 108);
113  data->scan(&entry->userId_, 8, 108, 116);
114  data->scan(&entry->groupId_, 8, 116, 124);
115  data->scan(&entry->size_, 8, 124, 136);
116  data->scan(&entry->lastModified_, 8, 136, 148);
117 
118  if (entry->type() == 0 && entry->path()->count() > 0) {
119  if (entry->path()->at(entry->path()->count() - 1) == '/')
120  entry->type_ = ArchiveEntry::Directory;
121  }
122 
123  entry->offset_ = i_;
124 
125  return true;
126 }
