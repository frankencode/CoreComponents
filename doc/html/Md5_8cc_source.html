30 void Md5::write(const ByteArray *data)
31 {
32  const uint8_t *src = data->bytes();
33  int srcLeft = data->count();
34  while (srcLeft > 0) {
35  while ((auxFill_ < aux_->count()) && (srcLeft > 0)) {
36  int n = aux_->count() - auxFill_;
37  if (n > srcLeft) n = srcLeft;
38  memcpy(aux_->bytes() + auxFill_, src, n);
39  src += n;
40  srcLeft -= n;
41  auxFill_ += n;
42  bytesFeed_ += n;
43  }
44  if (auxFill_ % 64 == 0)
45  consume();
46  }
47 }

49 Ref<ByteArray> Md5::finish()
50 {
51  /** manually feed the padding and message size
52  */
53  aux_->byteAt(auxFill_++) = 0x80;
54  while ((auxFill_ % 64) != 56)
55  aux_->byteAt(auxFill_++) = 0x00;
56  uint64_t b = bytesFeed_ << 3; // message length in bits
57  for (int i = 0; i < 8; ++i)
58  aux_->byteAt(auxFill_++) = b >> (i * 8);
59  consume();
60 
61  /** serialize the message digest
62  */
63  Ref<ByteArray> h = ByteArray::create(Md5::Size);
64  int k = 0;
65  for (int i = 0; i < 4; ++i) h->at(k++) = a_ >> (8 * i);
66  for (int i = 0; i < 4; ++i) h->at(k++) = b_ >> (8 * i);
67  for (int i = 0; i < 4; ++i) h->at(k++) = c_ >> (8 * i);
68  for (int i = 0; i < 4; ++i) h->at(k++) = d_ >> (8 * i);
69  return h;
70 }

175 void Md5::consume()
176 {
177  uint32_t *m = aux_->words();
178  for (int i = 0, nc = auxFill_ / 64; i < nc; ++i) {
179  // select 16 dwords x[0], x[1] ... x[15] and
180  // ensure value of the dwords is independent from system endianess
181  uint32_t *x = m + i * 16;
182  if (localEndian() == BigEndian) {
183  for (int j = 0; j < 16; ++j)
184  x[j] = swap32(x[j]);
185  }
186  // load state
187  uint32_t a = a_;
188  uint32_t b = b_;

191  // round 1
