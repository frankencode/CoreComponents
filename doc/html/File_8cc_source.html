101 String File::map() const
102 {
103  off_t fileEnd = ::lseek(fd_, 0, SEEK_END);
104  if (fileEnd == -1)
105  CC_SYSTEM_ERROR(errno, path_);
106  size_t fileSize = fileEnd;
107  if (fileSize == 0) return "";
108  if (fileSize >= size_t(intMax)) fileSize = intMax;
109  int pageSize = System::pageSize();
110  size_t mapSize = fileSize;
111  int protection = PROT_READ | (PROT_WRITE * (openFlags_ & (O_WRONLY|O_RDWR)));
112  void *p = 0;
113  if (fileSize % pageSize > 0) {
114  mapSize += pageSize - fileSize % pageSize;
115  p = ::mmap(0, fileSize, protection, MAP_PRIVATE, fd_, 0);
116  if (p == MAP_FAILED)
117  CC_SYSTEM_ERROR(errno, path_);
118  }
119  else {
120  #ifndef MAP_ANONYMOUS
121  #define MAP_ANONYMOUS MAP_ANON
122  #endif
123  mapSize += pageSize;
124  p = ::mmap(0, mapSize, PROT_READ, MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
125  if (p == MAP_FAILED)
126  CC_SYSTEM_DEBUG_ERROR(errno);
127  p = ::mmap(p, fileSize, protection, MAP_PRIVATE | MAP_FIXED, fd_, 0);
128  if (p == MAP_FAILED)
129  CC_SYSTEM_ERROR(errno, path_);
130  }
131  #ifdef MADV_SEQUENTIAL
132  if (::madvise(p, mapSize, MADV_SEQUENTIAL) == -1)
133  CC_SYSTEM_DEBUG_ERROR(errno);
134  #endif
135  return String(
136  Ref<ByteArray>(
137  new MappedByteArray(
138  reinterpret_cast<char *>(p),
139  fileSize
140  )
141  )
142  );
143 }

214  String buf = String(128);
215  while (true) {
216  ssize_t numBytes = ::readlink(path, buf, buf->count());
217  if (numBytes == -1)
218  return String();

242 String File::createUnique(String path, int mode, char placeHolder)
243 {
244  Ref<Random> random = Random::open(Process::currentId());
245  while (true) {
246  String candidate = path->copy();
247  for (int i = 0, n = candidate->count(); i < n; ++i) {
248  if (candidate->at(i) == placeHolder) {
249  char r = random->get(0, 61);
250  if ((0 <= r) && (r <= 9))
251  r += '0';
252  else if ((10 <= r) && (r <= 35))
253  r += 'a' - 10;
254  else if ((36 <= r) && (r <= 61))
255  r += 'A' - 36;
256  candidate->at(i) = r;
257  }
258  }
259  int fd = ::open(candidate, O_RDONLY|O_CREAT|O_EXCL, mode);
260  if (fd == -1) {
261  if (errno != EEXIST)
262  CC_SYSTEM_RESOURCE_ERROR(errno, candidate);
263  }
264  else {
265  ::close(fd);
266  return candidate;
267  }
268  }
269 }
