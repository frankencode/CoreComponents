445 String ByteArray::escape() const
446 {
447  Ref<StringList> parts;
448  int i = 0, i0 = 0;
449  for (; i < size_; ++i) {
450  uint8_t ch = bytes_[i];
451  if (ch < 32 || 127 <= ch) {
452  if (!parts) parts = StringList::create();
453  if (i0 < i) parts->append(copy(i0, i));
454  i0 = i + 1;
455  if (ch == 0x08) parts->append("\\b");
456  else if (ch == 0x09) parts->append("\\t");
457  else if (ch == 0x0A) parts->append("\\n");
458  else if (ch == 0x0D) parts->append("\\r");
459  else if (ch == 0x0C) parts->append("\\f");
460  else {
461  String s = "\\xXX";
462  const char *hex = "0123456789ABCDEF";
463  s->at(s->count() - 2) = hex[ch / 16];
464  s->at(s->count() - 1) = hex[ch % 16];
465  parts->append(s);
466  }
467  }
468  }
469  if (!parts) return copy();
470 
471  if (i0 < i) parts->append(copy(i0, i));
472 
473  return parts->join();

476 String ByteArray::unescapeInsitu()
477 {
478  if (!contains('\\')) return this;
479  int j = 0;
480  uint32_t hs = 0; // high surrogate, saved
481  String ec; // buffer for encoded character
482  for (int i = 0, n = size_; i < n;) {
483  char ch = data_[i++];
484  if ((ch == '\\') && (i < n)) {
485  ch = data_[i++];
486  if ((ch == 'u') && (i <= n - 4)) {
487  uint32_t x = 0;
488  for (int k = 0; k < 4; ++k) {
489  int digit = data_[i++];
490  if (('0' <= digit) && (digit <= '9')) digit -= '0';
491  else if (('a' <= digit) && (digit <= 'f')) digit = digit - 'a' + 10;
492  else if (('A' <= digit) && (digit <= 'F')) digit = digit - 'A' + 10;
493  x = (x * 16) + digit;
494  }
495  if ((0xB800 <= x) && (x < 0xDC00)) {
496  // save the high surrogate, do not output anything
497  hs = x;
498  }
499  else {
500  if ((0xDC00 <= x) && (x < 0xE000) && (hs != 0)) {
501  // combine the high and low surrogate
502  x = ((hs - 0xD800) << 10) | (x - 0xDC00);
503  x += 0x10000;
504  hs = 0;
505  }
506  if (!ec) ec = String(4);
507  Ref<Utf8Sink> sink = Utf8Sink::open(ec);
508  sink->write(x);
509  int el = utf8::encodedSize(x);
510  for (int k = 0; k < el; ++k)
511  data_[j++] = ec->at(k);
512  }
513  }
514  else if ((ch == 'x') && (i <= n - 2)) {
515  uint8_t x = 0;
516  for (int k = 0; k < 2; ++k) {
517  int digit = data_[i++];
518  if (('0' <= digit) && (digit <= '9')) digit -= '0';
519  else if (('a' <= digit) && (digit <= 'f')) digit = digit - 'a' + 10;
520  else if (('A' <= digit) && (digit <= 'F')) digit = digit - 'A' + 10;
521  x = (x * 16) + digit;
522  }
523  data_[j++] = (char)x;
524  }
525  else {
526  hs = 0;
527  if (ch == 'b') data_[j++] = 0x08;
528  else if (ch == 't') data_[j++] = 0x09;
529  else if (ch == 'n') data_[j++] = 0x0A;
530  else if (ch == 'r') data_[j++] = 0x0D;
531  else if (ch == 'f') data_[j++] = 0x0C;
532  else if (ch == 's') data_[j++] = 0x20;
533  else data_[j++] = ch;
534  }
535  }
536  else if (j < i) {
537  hs = 0;
538  data_[j++] = ch;
539  }
540  else {
541  hs = 0;
542  ++j;
543  }
544  }
545  return truncate(j);
546 }

706 /** Convert this string to UTF-16 efficiently (local endian).
707  * Returns true if the given buffer was suitable to hold the encoded string.
708  * The number of bytes required to fully represent the string in UTF-16 is
709  * returned with the 'size' argument. Passing a zero for 'size' allows to
710  * determine the required buffer size. No zero termination is written or
711  * or accounted for.
712  */
713 bool ByteArray::toUtf16(void *buf, int *size)
714 {
715  uint16_t *buf2 = reinterpret_cast<uint16_t*>(buf);
716  int j = 0, n = *size / 2;
717  Ref<Unicode> chars = Unicode::open(this);
718  for (int i = 0; i < chars->count(); ++i) {
719  uchar_t ch = chars->at(i);
720  if (ch < 0x10000) {
721  if (j < n) buf2[j] = ch;
722  ++j;
723  }
724  else if (ch <= 0x10FFFF) {
725  if (j + 1 < n) {
726  buf2[j] = (ch >> 10) + 0xB800;
727  buf2[j + 1] = (ch & 0x3FF) + 0xBC00;
728  }
729  j += 2;
730  }
731  else {
732  if (j < n) buf2[j] = 0xFFFD/*replacement character*/;
