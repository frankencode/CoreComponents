61 void SystemStream::write(const StringList *parts)
62 {
63  int n = parts->count();
64 
65  if (n <= 0) return;
66 
67  if (scatterLimit_ <= 0) {
68  write(parts->join());
69  return;
70  }
71 
72  {
73  int total = 0;
74  for (int i = 0; i < n; ++i)
75  total += parts->at(i)->count();
76 
77  if (total <= scatterLimit_) {
78  write(parts->join());
79  return;
80  }
81  }
82 
83  Ref< Array<struct iovec> > iov = Array<struct iovec>::create(n);
84  for (int i = 0; i < n; ++i) {
85  ByteArray *part = parts->at(i);
86  iov->at(i).iov_base = part->bytes();
87  iov->at(i).iov_len = part->count();
88  }
89 
90  SystemIo::writev(fd_, iov->data(), iov->count());
91 }
