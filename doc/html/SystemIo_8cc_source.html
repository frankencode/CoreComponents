23 bool SystemIo::poll(int fd, int events, int timeout_ms)
24 {
25  struct pollfd fds;
26  fds.fd = fd;
27  fds.events = events;
28 
29  if (timeout_ms < 0) timeout_ms = -1;
30  int ret = ::poll(&fds, 1, timeout_ms);
31  if (ret == -1) CC_SYSTEM_DEBUG_ERROR(errno);
32  CC_ASSERT(ret == 0 || ret == 1);
33  CC_ASSERT(timeout_ms != -1 || ret == 1);
34 
35  return ret == 1;
36 }

58 void SystemIo::writev(int fd, const struct iovec *iov, int iovcnt)
59 {
60  const int iovMax = SystemIo::instance()->iovMax();
61  for (int i = 0; i < iovcnt;) {
62  int m = iovcnt - i;
63  if (m > iovMax) m = iovMax;
64  ssize_t ret = ::writev(fd, iov + i, m);
