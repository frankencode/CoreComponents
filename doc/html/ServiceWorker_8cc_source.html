42 void ServiceWorker::logDelivery(HttpClientConnection *client, int statusCode, size_t bytesWritten, const String &statusMessage)
43 {
44  Stream *stream = accessLog()->noticeStream();
45  if (400 <= statusCode && statusCode <= 499) stream = accessLog()->debugStream();
46  else if (500 <= statusCode) stream = accessLog()->errorStream();
47 
48  HttpRequest *request = client->request();
49  String requestHost = request ? request->host() : "";
50  String requestLine = request ? request->line() : "";
51  double requestTime = request ? request->time() : System::now();
52  String userAgent = request ? request->value("User-Agent") : statusMessage;
53 
54  Format(stream)
55  << client->address()->networkAddress() << " "
56  << Date::create(requestTime)->toString() << " "
57  << "\"" << requestHost << "\" "
58  << "\"" << requestLine << "\" "
59  << statusCode << " " << bytesWritten << " "
60  << "\"" << userAgent << "\" "
61  << client->priority()
62  << nl;
63 }

65 void ServiceWorker::run()
66 {
67  errorLog()->open(nodeConfig()->errorLogConfig());
68  accessLog()->open(nodeConfig()->accessLogConfig());
69 
70  while (true) {
71  try {
72  if (!pendingConnections_->popFront(&client_)) break;
73 
74  int requestCount = 0;
75  try {
76  serviceInstance_ = client_->handshake();
77  if (!serviceInstance_) throw BadRequest();
78 
79  serviceDelegate_ = serviceInstance_->createDelegate(this);
80 
81  while (client_) {
82  CCNODE_DEBUG() << "Reading request..." << nl;
83  Ref<HttpRequest> request = client_->readRequest();
84  ++requestCount;
85  {
86  RefGuard<HttpResponse> guard(&response_);
87  response_ = HttpResponse::create(client_);
88 
89  /*if (requestCount == serviceInstance_->requestLimit()) {
90  response_->setHeader("Connection", "close");
91  }
92  else if (requestCount == 1) { FIXME: obsolete
93  response_->setHeader(
94  "Keep-Alive",
95  Format("timeout=%%, max=%%")
96  << int(nodeConfig()->connectionTimeout())
97  << serviceInstance_->requestLimit()
98  );
99  }*/
100 
101  serviceDelegate_->process(request);
102  response_->endTransmission();
103  if (response_->delivered()) {
104  logDelivery(client_, response_->statusCode(), response_->bytesWritten());
105  if (!client_->isPayloadConsumed())
106  break;
107  }
108  else break;
109  }
110  if (
111  request->value("Connection")->equalsCaseInsensitive("close") ||
112  requestCount >= serviceInstance_->requestLimit() ||
113  (request->majorVersion() == 1 && request->minorVersion() == 0)
114  ) break;
115  }
116  }
117  catch (ProtocolException &ex) {
118  if (requestCount > 0 || ex.statusCode() == RequestTimeout::StatusCode) {
119  try {
120  Format("HTTP/1.1 %% %%\r\n\r\n", client_->stream()) << ex.statusCode() << " " << ex.message();
121  }
122  catch(...)
123  {}
124  logDelivery(client_, ex.statusCode());
125  }
126  }
127  catch (Exception &ex) {
128  CCNODE_ERROR() << ex.message() << nl;
129  }
130  catch (CloseRequest &)
131  {}
132 
133  closeConnection();
134 
135  serviceDelegate_ = 0;
136  serviceInstance_ = 0;
137  }
138  catch (ProtocolException &ex)
139  {}
140  catch (ConnectionResetByPeer &)
141  {}
142  catch (Exception &ex) {
143  CCNODE_ERROR() << ex.message() << nl;
144  }
145  }
146 }

153 void ServiceWorker::autoSecureForwardings()
154 {
155  if (response()->statusCode() == 302) {
156  HttpResponse::Header *header = response()->header_;
157  if (
158  nodeConfig()->security()->hasCredentials() ||
159  serviceInstance_->security()->hasCredentials()
160  ) {
161  String location = header->value("Location");
162  if (location->beginsWith("http:")) {
163  try {
164  Ref<Uri> uri = Uri::parse(location);
165  if (serviceInstance_->host()->match(uri->host())) {
166  uri->setScheme("https");
167  if (uri->port() > 0)
168  uri->setPort(nodeConfig()->securePort());
169  header->establish("Location", uri->toString());
170  }
171  }
172  catch (...)
173  {}
174  }
175  }
176  }
177 }
