15 /* Every 4th year is a leap year excluding every last year in the
16  * 100 year cycle (century) but not the last year in a 400 year cycle.
17  */
18 inline bool leapYear(int y)
19 {
20  return (y % 4 == 0) && ((y % 100 != 0) || (y % 400 == 0));
21 }
22 
23 /** Number of leap years between x and y (excluding x, y).
24  */
25 int leaps(int x, int y)
26 {
27  if (x > y) {
28  int h = x;
29  x = y;
30  y = h;
31  }
32  --y; ++x; // exclude year x, y
33  if (x > y) return 0;
34  return y / 4 - y / 100 + y / 400 - (x - 1) / 4 + (x - 1) / 100 - (x - 1) / 400;
35 }

37 /** Days in a year are arbitrarily distributed but weighted to make the
38  * average length of a year match the length of the solar year.
39  * Interestingly the length of the year is measured in days, which is a duration
40  * depending on Earth's angular velocity.
41  */
42 inline int daysInYear(int y)
43 {
44  return 31 * 7 + 30 * 4 + 28 + leapYear(y);
45 }

47 inline int daysInMonth(int i, int y)
48 {
49  const int days[] = {
50  31/*Jan*/, 28/*Feb*/, 31/*Mar*/, 30/*Apr*/, 31/*May*/, 30/*Jun*/,
51  31/*Jul*/, 31/*Aug*/, 30/*Sep*/, 31/*Oct*/, 30/*Nov*/, 31/*Dec*/
52  };
53  return days[i] + (i == 1) * leapYear(y);

74 Date::Date(double time, int offset):
75  offset_(offset),
76  time_(time)
77 {
78  const int C1 = 365;
79  const int C4 = 4 * C1 + 1;
80  const int C100 = 25 * C4 - 1;
81  const int C400 = 4 * C100 + 1;
82  const int N0 = 719528; // linear day number of Jan 1st year 0
83 
84  int64_t t = int64_t(time) + int64_t(N0) * 86400;
85  int n = t / 86400;
86  if (n < 0) n = 0;
87 
88  hour_ = (t / 3600) % 24;
89  minutes_ = (t / 60) % 60;
90  seconds_ = t % 60;
91  weekDay_ = (n + 6) % 7;
92 
93  int y = 400 * (n / C400);
94  n = n % C400;
95 
96  while (true) {
97  int h = C100 + leapYear(y);
98  if (n < h) break;
99  n -= h;
100  y += 100;
101  }
102 
103  while (true) {
104  int h = C4 - !leapYear(y);
105  if (n < h) break;
106  n -= h;
107  y += 4;
108  }
109 
110  while (true) {
111  int h = C1 + leapYear(y);
112  if (n < h) break;
113  n -= h;
114  ++y;
115  }
116 
117  year_ = y;
118  yearDay_ = n;
119 
120  month_ = 0;
121  while (true) {
122  int h = daysInMonth(month_, y);
123  if (n < h) break;
124  ++month_;
125  n -= h;
126  }
127  ++month_;
128  day_ = n + 1;
129 }

131 Date::Date(int year, int month, int day, int hour, int minutes, int seconds, int offset):
132  year_(year),
133  month_(month),
134  day_(day),
135  hour_(hour),
136  minutes_(minutes),
137  seconds_(seconds),
138  offset_(offset)
139 {
140  if (year < 1) year = 1;
141  if (month > 12) month = 12;
142  else if (month < 1) month = 1;
143  if (day > 31) day = 31;
144  else if (day < 1) day = 1;
145  if (hour < 0) hour = 0;
146  else if (hour > 23) hour = 23;
147  if (minutes > 59) minutes = 59;
148  else if (minutes < 0) minutes = 0;
149  if (seconds > 59) seconds = 59;
150  else if (seconds < 0) seconds = 0;
151  if (offset > 600) offset = 600;
152  else if (offset < -600) offset = -600;
153 
154  --day;
155  --month;
156 
157  int64_t t = 0;
158  for (int i = 0; i < month; ++i)
159  t += daysInMonth(i, year);
160  t += day;
161  yearDay_ = t;
162  t += (31 * 7 + 30 * 4 + 28) * (year - 1970);
163  if (year >= 1970)
164  t += leaps(1970, year);
165  else
166  t -= leaps(1970, year - 1);
167  weekDay_ = (719528 + t + 6) % 7;
168  t *= 86400;
169  t += 3600 * hour + 60 * minutes + seconds;
170  time_ = t;
171 }
