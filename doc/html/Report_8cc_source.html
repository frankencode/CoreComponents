23 Report::Report(StringList *dirPaths, RegExp works, int worksMinLines):
24  dirPaths_(dirPaths),
25  works_(works),
26  worksMinLines_(worksMinLines),
27  coverage_(Coverage::create()),
28  exposure_(Exposure::create()),
29  coverageByDigest_(CoverageByDigest::create()),
30  coverageByHolder_(CoverageByHolder::create()),
31  statementByDigest_(StatementByDigest::create())
32 {
33  for (int i = 0; i < dirPaths->count(); ++i) {
34  String dirPath = dirPaths->at(i)->canonicalPath();
35  Ref<DirWalker> dirWalker = DirWalker::open(dirPath);
36  dirWalker->setIgnoreHidden(true);
37  for (String path; dirWalker->read(&path);) {
38  if (FileStatus::read(path)->type() != RegularFileType) continue;
39  if (!works->match(path->fileName())->valid()) continue;
40  Ref<Notice> notice;
41  {
42  for (int j = 0; j < registry()->headerStyleCount(); ++j) {
43  Ref<Header> header = registry()->headerStyleAt(j)->scan(path);
44  if (header) {
45  notice = noticeParser()->readNotice(header);
46  if (notice) break;
47  }
48  }
49  }
50  if (notice) {
51  coverage_->insert(path, notice);
52  {
53  String digest = cc::crypto::sha1(notice->statement());
54  Ref<Coverage> coverage;
55  if (!coverageByDigest_->lookup(digest, &coverage)) {
56  coverageByDigest_->insert(digest, coverage = Coverage::create());
57  statementByDigest_->insert(digest, notice->statement());
58  }
59  coverage->insert(path, notice);
60  }
61  {
62  CopyrightList *copyrights = notice->copyrights();
63  for (int j = 0; j < copyrights->count(); ++j) {
64  String holder = copyrights->at(j)->holder();
65  Ref<Coverage> coverage;
66  if (!coverageByHolder_->lookup(holder, &coverage))
67  coverageByHolder_->insert(holder, coverage = Coverage::create());
68  coverage->insert(path, notice);
69  }
70  }
71  }
72  else {
73  bool omit = false;
74  if (worksMinLines > 1) {
75  String text = File::open(path)->map();
76  int i = -1, n = 1;
77  for (; n < worksMinLines; ++n) {
78  i = text->find('\n', i + 1);
79  if (i == text->count()) break;
80  }
81  omit = (n < worksMinLines);
82  }
83  if (!omit)
84  exposure_->insert(path);
85  }
86  }
87  }
88 }
