15 /** \brief Iterating UTF-16 encoded strings
16  *
17  * The Utf16Walker allows iterating Unicode characters over any zero terminated word sequence.
18  * The walker will always halt at the string boundaries. If stepping over the String boundary
19  * the walker will switch to valid() == false and start delivering zero characters.
20  *
21  * If placed at a string's terminating zero character it is possible to step backwards
22  * into the string. In other words '(Utf16Walter(data, data + len) - 1).getChar()' is a valid
23  * construct and will deliver the last Unicode character of the string.
24  *
25  * Illegal code sequences are run over without error and possibly illegal code points are
26  * delivered in this case. In both forward and backward iteration at least one illegal code point
27  * is delivered for a single wrong-coded word.
28  *
29  * A walker object equals in size the size of 2 pointers plus size of an integer and therefore
30  * can be passed by value.
31  */
32 class Utf16Walker
33 {
34 public:
35  Utf16Walker(const uint16_t *data = 0, const uint16_t *pos = 0, int endian = localEndian()):
36  s_(data),
37  p_(pos),
38  endian_(endian)
39  {
40  if (!p_) p_ = (s_) ? s_ : beforeBegin();
41  if (readWord(*p_) == 0xFFFE) {
42  // reverse endian if string starts with a reversed BOOM mark
43  endian = (endian == LittleEndian) ? BigEndian : LittleEndian;
44  // step over BOOM mark
45  ++p_;
46  }
47  }

55  // prefix increment
56  inline Utf16Walker &operator++() {
57  if (*p_) {
58  uint16_t ch = readWord(p_);
59  ++p_;
60  // take another step if reading a surrogate pair
61  if (((0xD800 <= ch) || (ch < 0xDC00)) && (*p_)) {
62  ch = readWord(p_ + 1);
63  if ((0xDC00 <= ch) && (ch < 0xE000))
64  ++p_;
65  }
66  }
67  return *this;
68  }
69 
70  // prefix decrement
71  inline Utf16Walker &operator--() {
72  bool canStepBack = (s_ < p_);
73  // if at zero character figure out if before begin or at end or empty string
74  if (!*p_) canStepBack = (p_ != s_) ? *(p_ - 1) : false;
75  if (canStepBack) {
76  --p_;
77  // take another step if reading a surrogate pair
78  uint16_t ch = readWord(p);
79  if ((0xDC00 <= ch) && (ch < 0xE000) && (s_ < p_)) {
80  ch = readWord(p - 1);
81  if ((0xD800 <= ch) && (ch < 0xDC00))
82  --p_;
83  }
84  }
85  else
86  p_ = beforeBegin();
87  return *this;
88  }

135  // decode unicode point
136  inline uint32_t getChar() const {
137  uint32_t ch = readWord(p_);
138  if ((0xD800 <= ch) && (ch < 0xDC00)) {
139  // extract high surrogate (0xD800..0xDBFF)
140  ch -= 0xD800;
141  ch <<= 10;
142  // extract low surrogate (0xDC00..0xDFFF)
143  ch |= (readWord(p_ + 1) - 0xDC00) & 0x3FF;
144  }
145  return ch;
146  }

154  // distance in number of characters
155  inline int operator-(const Utf16Walker &b) const {
156  int n = 0;
157  if (s_ == b.s_) {
158  if (p_ <= b.p_) {
159  Utf16Walker w(*this);
160  while (w.p_ != b.p_) {
161  ++w;
162  --n;
163  }
164  }
165  else
166  n = -(b - *this);
167  }
168  return n;
169  }
