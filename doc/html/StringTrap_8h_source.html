36  inline const char *capture(const char *s, int len = -1) const {
37  if (len < 0) len = strlen(s);
38  if (!head_) head_ = tail_ = new Node(s, len);
39  else tail_ = tail_->next_ = new Node(s, len);
40  return tail_->s_->chars();
41  }
