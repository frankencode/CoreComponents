14 const char *Variant::typeName(int type, int itemType)
15 {
16  if (type == UndefType) return "undefined";
17  else if (type == IntType) return "integer";
18  else if (type == BoolType) return "boolean";
19  else if (type == FloatType) return "real";
20  else if (type == ColorType) return "color";
21  else if (type == VersionType) return "version";
22  else if (type == ObjectType) return "object";
23  else if (type == StringType) return "string";
24  else if (type == AnyType) return "any";
25  else if (type == ListType) {
26  if (itemType == IntType) return "list of integer";
27  else if (itemType == BoolType) return "list of boolean";
28  else if (itemType == FloatType) return "list of real";
29  else if (itemType == ColorType) return "list of color";
30  else if (itemType == VersionType) return "list of version";
31  else if (itemType == ObjectType) return "list of object";
32  else if (itemType == StringType) return "list of string";
33  else return "list";
34  }
35  else if (type == MapType) {
36  return "map";
37  }
38  return "unknown";
39 }
