68 Ref<Date> scanDate(String text, bool *ok)
69 {
70  // e.g.: Tue, 10 Sep 2013 11:01:10 GMT
71  Ref<StringList> parts = text->split(' ');
72  if (ok) *ok = false;
73  if (parts->count() != 6) return 0;
74  int day = 0;
75  {
76  String s = parts->at(1);
77  for (int i = 0; i < s->count(); ++i) {
78  char ch = s->at(i);
79  day *= 10;
80  if ('0' <= ch && ch <= '9') day += ch - '0';
81  else return 0;
82  }
83  if (day > 31) return 0;
84  }
85  int month = 0;
86  {
87  const char *names[] = { "Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec" };
88  String s = parts->at(2);
89  for (; month < 12; ++month) {
90  if (s == names[month]) break;
91  }
92  if (month > 11) return 0;
93  ++month;
94  }
95  int year = 0;
96  {
97  String s = parts->at(3);
98  for (int i = 0; i < s->count(); ++i) {
99  char ch = s->at(i);
100  year *= 10;
101  if ('0' <= ch && ch <= '9') year += ch - '0';
102  else return 0;
103  }
104  }
105  int hour = 0, minutes = 0, seconds = 0;
106  {
107  String s = parts->at(4);
108  int *p = &hour;
109  for (int i = 0; i < s->count(); ++i) {
110  char ch = s->at(i);
111  if (ch == ':') {
112  if (p == &hour) p = &minutes;
113  else if (p == &minutes) p = &seconds;
114  else return 0;
115  }
116  else {
117  *p *= 10;
118  if ('0' <= ch && ch <= '9') *p += ch - '0';
119  else return 0;
120  }
121  }
122  }
123  if (ok) *ok = true;
124  return Date::create(year, month, day, hour, minutes, seconds);
125 }
