18 void dump(String prefix, ByteArray *data)
19 {
20  Ref<StringList> parts = data->split('\n');
21  for (int i = 0; i < parts->count(); ++i) {
22  if (i != parts->count() - 1)
23  parts->at(i) = parts->at(i) + "\n";
24  if (parts->at(i)->count() == 0) continue;
25  parts->at(i) = prefix + " " + parts->at(i)->escape() + "\n";
26  }
27  fout() << parts->join();
28  ferr() << "-----\n";
29 }
