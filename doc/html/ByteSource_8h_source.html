163 inline uint8_t ByteSource::readUInt8()
164 {
165  if (i_ == n_) {
166  if (!source_) throw UnexpectedEndOfInputError();
167 
168  n_ = source_->read(buffer_);
169  i_ = 0;

172  }
173 
174  return buffer_->at(i_++);

177 inline uint16_t ByteSource::readUInt16()
178 {
179  uint16_t h;
180  if (endian_ == LittleEndian)
181  {
182  h = uint16_t(readUInt8());
183  h |= uint16_t(readUInt8()) << 8;
184  }
185  else // if (endian_ == BigEndian)
186  {
187  h = uint16_t(readUInt8()) << 8;
188  h |= uint16_t(readUInt8());
189  }
190  return h;
191 }

213 inline uint64_t ByteSource::readUInt64()
214 {
215  uint64_t h = 0;
216  if (endian_ == LittleEndian)
217  {
218  for (int i = 0; i <= 7*8; i += 8)
219  h |= uint64_t(readUInt8()) << i;
220  }
221  else // if (endian_ == BigEndian)
222  {
223  for (int i = 7*8; i >= 0; i -= 8)
224  h |= uint64_t(readUInt8()) << i;
225  }
226  return h;
