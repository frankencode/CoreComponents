80 void NodeMaster::runNode() const
81 {
82  CCNODE_NOTICE() << "Starting (pid = " << Process::currentId() << ")" << nl;
83 
84  if (nodeConfig()->directoryPath() != "") {
85  ServiceDefinition *directoryService = serviceRegistry()->serviceByName("Directory");
86  MetaObject *config = directoryService->configPrototype();
87  config->establish("host", "*");
88  config->establish("path", nodeConfig()->directoryPath());
89  Ref<ServiceInstance> directoryInstance = directoryService->createInstance(config);
90  nodeConfig()->serviceInstances()->append(directoryInstance);
91  }
92 
93  if (nodeConfig()->serviceInstances()->count() == 0)
94  {
95  CCNODE_WARNING() << "No service configured, falling back to Echo service" << nl;
96 
97  ServiceDefinition *echoService = serviceRegistry()->serviceByName("Echo");
98  MetaObject *config = echoService->configPrototype();
99  config->establish("host", "*");
100  Ref<ServiceInstance> echoInstance = echoService->createInstance(config);
101  nodeConfig()->serviceInstances()->append(echoInstance);
102  }

104  typedef List< Ref<HttpSocket> > ListeningSockets;
105  Ref<ListeningSockets> listeningSockets = ListeningSockets::create(nodeConfig()->address()->count());
106 
107  for (int i = 0; i < nodeConfig()->address()->count(); ++i) {
108  SocketAddress *address = nodeConfig()->address()->at(i);
109  CCNODE_NOTICE() << "Start listening at " << address << nl;
110  listeningSockets->at(i) = HttpSocket::listen(address);
111  }
112 
113  if (nodeConfig()->user() != "") {
114  String userName = nodeConfig()->user();
115  String groupName = nodeConfig()->group();
116  if (groupName == "") groupName = userName;
117  Ref<User> user = User::lookup(userName);
118  Ref<Group> group = Group::lookup(groupName);
119  if (!user->exists()) throw UsageError("No such user: \"" + userName + "\"");
120  if (!group->exists()) throw UsageError("No such group: \"" + groupName + "\"");
121  CCNODE_NOTICE() << "Dropping process persona to user:group " << userName << ":" << groupName << " (uid:gid = " << user->id() << ":" << group->id() << ")" << nl;
122  Process::setPersona(user->id(), group->id());
123  }
124 
125  CCNODE_NOTICE() << "Starting security master, if needed" << nl;
126  SecurityMaster::start();
127 
128  Ref<ConnectionManager> connectionManager = ConnectionManager::create();
129  Ref<PendingConnections> pendingConnections = PendingConnections::create();
130  Ref<ClosedConnections> closedConnections = connectionManager->closedConnections();
131 
132  CCNODE_NOTICE() << "Creating worker pool (concurrency = " << nodeConfig()->concurrency() << ")" << nl;
133 
134  typedef Array< Ref<ServiceWorker> > WorkerPool;
135  Ref<WorkerPool> workerPool = WorkerPool::create(nodeConfig()->concurrency());
136  for (int i = 0; i < workerPool->count(); ++i) {
137  Ref<ServiceWorker> worker = ServiceWorker::create(pendingConnections, closedConnections);
138  workerPool->at(i) = worker;
139  worker->start();
140  }
141 
142  CCNODE_NOTICE() << "Up and running (pid = " << Process::currentId() << ")" << nl;
143 
144  Ref<IoMonitor> ioMonitor = IoMonitor::create(listeningSockets->count());
145  for (int i = 0; i < listeningSockets->count(); ++i)
146  ioMonitor->addEvent(SystemIo::ReadyAccept, listeningSockets->at(i));
147 
148  CCNODE_DEBUG() << "Accepting connections" << nl;
149 
150  while (true) {
151  Ref<IoActivity> activity = ioMonitor->poll(1000);
152  if (activity->count() > 0) {
153  for (int i = 0; i < activity->count(); ++i) {
154  HttpSocket *socket = cast<HttpSocket>(activity->at(i)->target());
155  try {
156  Ref<HttpSocket> clientSocket = socket->accept();
157  Ref<HttpClientConnection> client = HttpClientConnection::open(clientSocket);
158  if (connectionManager->accept(client)) {
159  CCNODE_DEBUG() << "Accepted connection from " << client->address() << " with priority " << client->priority() << nl;
160  pendingConnections->pushBack(client, client->priority());
161  }
162  else {
163  CCNODE_DEBUG() << "Rejected connection from " << client->address() << nl;
164  }
165  }
166  catch (ProtocolException &ex) {
167  CCNODE_DEBUG() << ex.message() << nl;
168  }
169  catch (Exception &ex) {
170  CCNODE_DEBUG() << ex.message() << nl;
171  }
172  }
173  }
174 
175  connectionManager->cycle();
176 
177  while (signalMaster_->receivedSignals()->count() > 0) {
178  int signal = signalMaster_->receivedSignals()->popFront();
179  if (signal == SIGWINCH || signal == SIGPIPE) continue;
180  CCNODE_NOTICE() << "Received " << signalName(signal) << ", shutting down" << nl;
181  workerPool = 0;
182  CCNODE_NOTICE() << "Shutdown complete" << nl;
183  throw Interrupt(signal);
184  }
185  }
186 }
