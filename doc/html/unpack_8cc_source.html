42 void unpack(ArchiveReader *archive, bool verbose)
43 {
44  for (Ref<ArchiveEntry> entry; archive->readHeader(&entry);) {
45  if (verbose) fout() << entry->path() << nl;
46  if (entry->type() == ArchiveEntry::Directory) {
47  if (entry->path() == "." || entry->path() == "./") ;
48  else Dir::establish(entry->path(), entry->mode());
49  }
50  else if (entry->type() == ArchiveEntry::Link) {
51  File::link(entry->linkPath(), entry->path());
52  CC_SYSTEM_RESOURCE_ERROR(errno, entry->linkPath());
53  }
54  else if (entry->type() == ArchiveEntry::Symlink) {
55  File::symlink(entry->linkPath(), entry->path());
56  }
57  else if (entry->type() == ArchiveEntry::Regular) {
58  File::create(entry->path(), entry->mode());
59  if (Process::isSuperUser()) {
60  uid_t userId = entry->userId();
61  gid_t groupId = entry->groupId();
62  if (entry->userName() != "")
63  userId = User::lookup(entry->userName())->id();
64  if (entry->groupName() != "")
65  groupId = Group::lookup(entry->groupName())->id();
66  if (userId != 0 || groupId != 0)
67  File::chown(entry->path(), userId, groupId);
68  }
69  if (entry->size() > 0) {
70  Ref<File> file = File::open(entry->path(), File::WriteOnly);
71  archive->readData(entry, file);
72  }
73  Ref<FileStatus> status = FileStatus::read(entry->path());
74  status->setTimes(status->lastAccess(), entry->lastModified());
75  }
76  }
77 }
