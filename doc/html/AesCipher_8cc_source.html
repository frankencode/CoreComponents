147 inline uint8_t mul(uint8_t a, uint8_t b)
148 {
149  int c = 0, h = a, f = 1;
150  while (f <= b) {
151  if (b & f) c ^= h;
152  h = mul2(h);

154  }

158 inline void mixColumns(ByteArray *s_)
159 {
160  uint8_t *s = s_->bytes();
161 
162  for (int c = 0; c < Nb; ++c) {
163  uint8_t *sc = s + c * Nb;
164  uint8_t h0 = sc[0];
165  uint8_t h1 = sc[1];
166  uint8_t h2 = sc[2];
167  uint8_t h3 = sc[3];
168  sc[0] = h2 ^ h3 ^ mul2(h0) ^ mul3(h1);
169  sc[1] = h0 ^ h3 ^ mul2(h1) ^ mul3(h2);
170  sc[2] = h0 ^ h1 ^ mul2(h2) ^ mul3(h3);
171  sc[3] = h1 ^ h2 ^ mul2(h3) ^ mul3(h0);
172  }
173 }

232 inline uint8_t rCon(int i)
233 {
234  uint8_t h = 1;
235  for (int j = 0; j < i - 1; ++j) {
236  bool cf = h & 0x80;
237  h <<= 1;
238  if (cf) h ^= 0x1B;
239  }

243 Ref<ByteArray> keyExpansion(const ByteArray *key, int Nr)
244 {
245  if (Nr <= 0) Nr = numRounds(key->count() / 4);
246  Ref<ByteArray> w = ByteArray::create(Ns * (Nr + 1));
247 
248  const int Nk = key->count() / 4;
249  int i = 0;
250 
251  for (; i < Nk; ++i) {
252  w->wordAt(i) =
253  word(
254  key->at(4 * i),
255  key->at(4 * i + 1),
256  key->at(4 * i + 2),
257  key->at(4 * i + 3)
258  );
259  }
260 
261  for (; i < w->count() / 4; ++i) {
262  uint32_t h = w->wordAt(i - 1);
263  if (i % Nk == 0)
264  h = subWord(rotWord(h)) ^ rCon(i / Nk);
265  else if (Nk > 6 && i % Nk == 4)
266  h = subWord(h);
267  w->wordAt(i) = w->wordAt(i - Nk) ^ h;
268  }
269 
270  return w;
271 }

288 {

292  *s_ = *p;

294  addRoundKey(s_, w_, 0);

300  addRoundKey(s_, w_, r);
