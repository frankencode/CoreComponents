55 {
56  if (source->count() % 4 != 0) throw base64::IllegalInputSize4Error();
57 
58  const int m = source->count();
59  int p = 0;
60  while (m - p > 0) {
61  ++p;
62  if (source->at(m - p) != '=') {
63  --p;
64  break;
65  }
66  }
67 
68  if (!((0 <= p) && (p <= 2))) throw base64::IllegalPaddingError();
69 
70  int n = 3 * (m / 4) - p;
71  String sink(n);
72  int i = 0;
73 
74  for (int l = 0; l < m;) {
75  uint32_t bits = 0;
76  for (int k = 0; k < 4; ++k) {
77  uint32_t ch = source->at(l++);
78  if (('A' <= ch) && (ch <= 'Z')) ch -= 'A';
79  else if (('a' <= ch) && (ch <= 'z')) ch = (ch - 'a') + 26;
80  else if (('0' <= ch) && (ch <= '9')) ch = (ch - '0') + 52;
81  else if (ch == '+') ch = 62;
82  else if (ch == '/') ch = 63;
83  else if (ch == '=') ch = 0;
84  else throw base64::IllegalCharacterError();
85  bits |= ch;
86  bits <<= 6;
87  }
88  bits >>= 6;
89  for (int j = 0; j < 3; ++j) {
