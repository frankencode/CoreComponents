110  Ref<List> sort(int order = SortOrder::Ascending, bool unique = false) const
111  {
112  if (count() == 0)
113  return List::create();
114  Ref< Heap<Item> > heap = Heap<Item>::create(count(), order);
115  for (int i = 0; i < count(); ++i)
116  heap->push(at(i));
117  Ref<List> result;
118  if (unique) {
119  result = List::create();
120  Item prev, item;
121  heap->pop(&prev);
122  result->append(prev);
123  while (!heap->isEmpty()) {
124  heap->pop(&item);
125  if (item != prev) {
126  result->append(item);
127  prev = item;
128  }
129  }
130  }
131  else {
132  result = List::create(count());
133  for (int i = 0, n = result->count(); i < n; ++i)
134  result->at(i) = heap->pop();
135  }
136  return result;
137  }
