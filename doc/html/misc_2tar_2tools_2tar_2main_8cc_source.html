36  }
37  else {

39  }

42  }

44  Ref<VariantMap> options = arguments->options();
45  Ref<StringList> items = arguments->items();
46 
47  if (unpackMode) {
48  if (items->count() == 0) items->append(String());

50  for (int i = 0; i < items->count(); ++i) {
51  String path = items->at(i);
52 
53  Ref<Stream> source;
54  if (path != "") source = File::open(path);
55  else source = stdIn();
56

58  if (tarMode) archive = TarReader::open(source);
59  else archive = ArReader::open(source);
60 
61  if (options->value("list")) cctar::list(archive);
62  else if (options->value("status")) cctar::status(archive);
63  else cctar::unpack(archive, options->value("verbose"));
64  }
65  }
66  else {
67  Ref<Stream> sink;
68  String sinkPath = options->value("output");
69  if (sinkPath != "") {
70  try { File::unlink(sinkPath); } catch (SystemError &) {}
71  File::create(sinkPath);
72  sink = File::open(sinkPath, File::WriteOnly);
73  }
74  else {
75  sink = stdOut();
76  }
77 
78  Ref<ArchiveWriter> archive;
79  if (tarMode) archive = TarWriter::open(sink);
80  else archive = ArWriter::open(sink);
81 
82  for (int i = 0; i < items->count(); ++i)
83  cctar::pack(items->at(i), archive, options->value("verbose"));
84  }
85  }
86  catch (HelpError &) {
87  if (unpackMode) {
88  fout(

90  "Unpack or list contents of archive files.\n"
91  "\n"
92  "Options:\n"

94  " -status list archived file status\n"
95  " -verbose verbose output\n"
96  ) << toolName;
97  }
98  else {

101  "Pack an archive file.\n"
102  "\n"
103  "Options:\n"
104  " -output archive file to create\n"
105  " -verbose verbose output\n"
106  ) << toolName;
107  }
108  return 1;
109  }
110  catch (BrokenArchive &ex) {

112  return 1;
113  }
114  catch (Exception &ex) {

116  return 1;
117  }
