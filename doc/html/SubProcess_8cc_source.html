30 Ref<SubProcess> SubProcess::open(Params *params)
31 {
32  // ------------------------------------------------------------------------
33  // locate executable and prepare argument list
34  // ------------------------------------------------------------------------
35 
36  String execPath;
37  Ref<StringList> args = params->args_;
38 
39  if (params->command_ != "") {
40  String cmd = params->command_;
41  if (cmd->beginsWith(' ') || cmd->endsWith(' '))
42  cmd = cmd->trim();
43  if (!cmd->contains(' ')) execPath = cmd;
44  if (!args) args = cmd->simplify()->split(' ');
45  }
46 
47  if (args) if (args->count() == 0) args = 0;
48 
49  if (execPath == "" && args) execPath = args->at(0);
50 
51  if (execPath != "") {
52  if (execPath->contains('/')) {
53  if (!File::access(execPath, ExecuteOk)) throw CommandNotFound(execPath);
54  }
55  else {
56  String path = File::locate(execPath, Process::env("PATH")->split(':'), ExecuteOk);
57  if (path == "") throw CommandNotFound(execPath);
58  execPath = path;
59  }
60  }
61 
62  // ------------------------------------------------------------------------
63  // prepare I/O forwarding
64  // ------------------------------------------------------------------------
65 
66  enum { Master = 0, Slave = 1 };
67 
68  int fd[2];
69  fd[Master] = 0;
70  fd[Slave] = 0;
71 
72  if (params->forwarding_ == SubProcess::ForwardBySocket)
73  {
74  if (::socketpair(AF_LOCAL, SOCK_STREAM, 0, fd) == -1)
75  CC_SYSTEM_DEBUG_ERROR(errno);
76  }
77  else if (params->forwarding_ == SubProcess::ForwardByPts)
78  {
79  fd[Master] = ::posix_openpt(O_RDWR|O_NOCTTY);
80  if (fd[Master] == -1) CC_SYSTEM_DEBUG_ERROR(errno);
81  if (::grantpt(fd[Master]) == -1) CC_SYSTEM_DEBUG_ERROR(errno);
82  if (::unlockpt(fd[Master]) == -1) CC_SYSTEM_DEBUG_ERROR(errno);
83  fd[Slave] = ::open(ptsname(fd[Master]), O_RDWR|O_NOCTTY);
84  if (fd[Slave] == -1) CC_SYSTEM_DEBUG_ERROR(errno);
85  }
86 
87  // ------------------------------------------------------------------------
88  // prepare argument list and environment map
89  // ------------------------------------------------------------------------
90 
91  char **argv = 0;
92  char **envp = 0;
93 
94  if (execPath != "")
95  {
96  int argc = args ? args->count() : 1;
97  argv = new char*[argc + 1];
98 
99  if (args) {
100  for (int i = 0; i < args->count(); ++i)
101  argv[i] = cc::strdup(args->at(i)->chars());
102  }
103  else {
104  argv[0] = cc::strdup(execPath->chars());
105  }
106  argv[argc] = 0;
107 
108  EnvMap *envMap = params->envMap_;
109  if (envMap) {
110  int n = envMap->count();
111  envp = new char*[n + 1];
112  for (int i = 0; i < n; ++i)
113  envp[i] = cc::strcat(envMap->keyAt(i)->chars(), "=", envMap->valueAt(i)->chars());
114  envp[n] = 0;
115  }
116  }
117 
118  // ------------------------------------------------------------------------
119  // fork and setup child process
120  // ------------------------------------------------------------------------
121 
122  int ret = ::fork();
123  if (ret < 0) CC_SYSTEM_DEBUG_ERROR(errno);
124 
125  if (ret == 0)
126  {
127  if (params->type_ == SubProcess::GroupLeader)
128  ::setpgid(0, 0);
129  else if (params->type_ == SubProcess::SessionLeader)
130  ::setsid();
131 
132  if (params->forwarding_ == SubProcess::ForwardBySocket)
133  {
134  ::close(fd[Master]);
135  ::dup2(fd[Slave], 0);
136  ::dup2(fd[Slave], 1);
137  ::dup2(fd[Slave], 2);
138  }
139  else if (params->forwarding_ == SubProcess::ForwardByPts)
140  {
141  ::close(fd[Master]);
142  ::dup2(fd[Slave], 0);
143  ::dup2(fd[Slave], 1);
144  ::dup2(fd[Slave], 2);
145 
146  if (params->type_ == SubProcess::SessionLeader)
147  ::ioctl(fd[Slave], TIOCSCTTY, 0);
148 
149  {
150  struct termios attr;
151  ::tcgetattr(fd[Slave], &attr);
152  attr.c_iflag ^= IXON;
153  #ifdef IUTF8
154  attr.c_iflag |= IUTF8;
155  #endif // IUTF8
156  ::tcsetattr(fd[Slave], TCSANOW, &attr);
157  }
158  }
159 
160  if (params->overloads_) {
161  Overloads *overloads = params->overloads_;
162  for (int i = 0; i < overloads->count(); ++i)
163  ::dup2(overloads->valueAt(i)->fd(), overloads->keyAt(i));
164  }
165 
166  if (params->workDir_ != "")
167  ::chdir(params->workDir_);
168 
169  if (params->userMask_ >= 0)
170  ::umask(params->userMask_);
171 
172  if (params->signalMask_)
173  ::sigprocmask(SIG_SETMASK, params->signalMask_->rawSet(), 0);
174 
175  if (execPath != "") {
176  ::execve(execPath, argv, envp ? envp : ::environ);
177  ::exit(errno);
178  }
179 
180  ::exit(params->worker_ ? params->worker_->run() : 0);
181  }
182 
183  // ------------------------------------------------------------------------
184  // connect parent to child process
185  // ------------------------------------------------------------------------
186 
187  if (argv) {
188  for (int i = 0; argv[i]; ++i)
189  cc::free(argv[i]);
190  delete[] argv;
191  }
192  if (envp) {
193  for (int i = 0; envp[i]; ++i)
194  cc::free(envp[i]);
195  delete[] envp;
196  }
197 
198  Ref<SubProcess> child = new SubProcess(params, ret);
199 
200  if (
201  params->forwarding_ == SubProcess::ForwardBySocket ||
202  params->forwarding_ == SubProcess::ForwardByPts
203  ) {
204  ::close(fd[Slave]);
205  child->fd_ = fd[Master];
206  }
207 
208  return child;
209 }

277 int SubProcess::wait(bool *signalled)
278 {
279  int ret = 0;
280 
281  if (::waitpid(pid_, &ret, 0) == -1)
282  CC_SYSTEM_DEBUG_ERROR(errno);
283 
284  if (WIFEXITED(ret)) {
285  if (signalled) *signalled = false;
286  ret = WEXITSTATUS(ret);
287  }
288  else if (WIFSIGNALED(ret)) {
289  if (signalled) *signalled = true;
290  ret = WTERMSIG(ret);
291  }
292  else {
293  if (signalled) *signalled = false;
294  }
295 
296  pid_ = -1;
297 
298  return ret;
299 }
