28 DependencyCache::DependencyCache(BuildPlan *plan):
29  buildPlan_(plan),
30  cachePath_(cachePath(plan)),
31  cache_(Cache::create()),
32  previousSources_(StringList::create())
33 {
34  File::establish(cachePath_);
35  double cacheTime = FileStatus::read(cachePath_)->lastModified();
36 
37  Ref<MetaObject> dependencyCache;
38  try {
39  dependencyCache = yason::parse(File::load(cachePath_));
40  }
41  catch (TextError &)
42  {}
43  if (!dependencyCache) return;
44  if (dependencyCache->className() != "DependencyCache") return;
45 
46  for (int i = 0; i < dependencyCache->count(); ++i)
47  {
48  MetaObject::Item item = dependencyCache->at(i);
49 
50  previousSources_->append(item->key());
51 
52  if (!plan->sources()->contains(item->key())) continue;
53 
54  MetaObject *yason = cast<MetaObject>(item->value());
55  String command = yason->value("analyseCommand");
56  String modulePath = yason->value("modulePath");
57  Ref<StringList> dependencyPaths = cast<VariantList>(yason->value("dependencyPaths"))->toList<String>();
58  String sourcePath = dependencyPaths->at(0);
59 
60  bool dirty = false;
61 
62  Ref<FileStatus> objectStatus = buildPlan_->shell()->fileStatus(modulePath);
63  double objectTime = objectStatus->lastModified();
64 
65  for (int i = 0; i < dependencyPaths->count(); ++i)
66  {
67  Ref<FileStatus> sourceStatus = buildPlan_->shell()->fileStatus(dependencyPaths->at(i));
68  if (!sourceStatus->exists()) {
69  dirty = true;
70  break;
71  }
72  double sourceTime = sourceStatus->lastModified();
73  if (sourceTime > cacheTime) {
74  dirty = true;
75  break;
76  }
77  if (objectStatus->exists()) {
78  if (sourceTime > objectTime /*|| cacheItemTime > objectTime*/) {
79  dirty = true;
80  break;
81  }
82  }
83  }
84 
85  if (dirty) continue;
86 
87  if (command != buildPlan_->toolChain()->analyseCommand(plan, sourcePath)) continue;
88 
89  cache_->insert(
90  item->key(),
91  Module::create(command, modulePath, dependencyPaths, !objectStatus->exists())
92  );
93  }
94 }

96 DependencyCache::~DependencyCache()
97 {
98  Format text;
99  String indent(4, ' ');
100  text << "DependencyCache {\n";
101  for (int i = 0; i < cache_->count(); ++i) {
102  Cache::Item item = cache_->at(i);
103  String sourcePath = item->key();
104  Module *module = item->value();
105  text
106  << indent << "\"" << sourcePath << "\": Module {\n"
107  << indent << indent << "analyseCommand: \"" << module->analyseCommand() << "\"\n"
108  << indent << indent << "modulePath: \"" << module->modulePath() << "\"\n"
109  << indent << indent << "dependencyPaths: [\n";
110  for (int i = 0, n = module->dependencyPaths()->count(); i < n; ++i) {
111  text << indent << indent << indent << "\"" << module->dependencyPaths()->at(i);
112  if (i != n - 1)
113  text << "\",\n";
114  else
115  text << "\"\n";
116  }
117  text << indent << indent << "]\n"
118  << indent << "}\n";
119  }
120  text << "}\n";
121  File::save(cachePath_, text);
122 }
