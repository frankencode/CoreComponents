159 template<class Node>
160 bool OrdinalTree<Node>::lookupByIndex(int i, Node **node) const
161 {
162  CC_ASSERT((0 <= i) && (i < weight()));
163 
164  ExclusiveAccess cacheAccess(&cacheExclusive_);
165  if (cacheAccess) {
166  if (cachedNode_) {
167  const int d = i - cachedIndex_;
168  if (d == 1) {
169  ++cachedIndex_;
170  cachedNode_ = static_cast<Node *>(BinaryTree::succ(cachedNode_));
171  if ((cachedNode_) && (node)) *node = cachedNode_;
172  return cachedNode_;
173  }
174  else if (d == 0) {
175  if (node) *node = cachedNode_;
176  return cachedNode_;
177  }
178  else if (d == -1) {
179  --cachedIndex_;
180  cachedNode_ = static_cast<Node *>(BinaryTree::pred(cachedNode_));
181  if ((cachedNode_) && (node)) *node = cachedNode_;
182  return cachedNode_;
183  }
184  }
185  }
186 
187  Node *k = root_;
188  int j0 = 0;
189  while (k) {
190  int j = j0 + BinaryTree::weight(k->left_);
191  if (i < j) {
192  k = k->left();
193  }
194  else if (j < i) {
195  j0 = j + 1;
196  k = k->right();
197  }
198  else // i == j
199  break;
200  }
201  if ((k) && (node)) *node = k;
202 
203  if (cacheAccess) {
204  cachedNode_ = k;
205  cachedIndex_ = i;
206  }
207 
208  return k;
209 }

212 template<class Pattern>
213 Node *OrdinalTree<Node>::find(const Pattern &pattern, bool *found, bool *below, int *index) const
214 {
215  Node *k = root_;
216  Node *k2 = 0;
217  if (found) *found = false;
218  int j0 = 0, j = -1;
219  if (k) {
220  while (true) {
221  k2 = k;
222  j = j0 + BinaryTree::weight(k->left_);
223  if (pattern < k->item_) {
224  if (!k->left_) {
225  if (below) *below = true;
226  break;
227  }
228  k = k->left();
229  }
230  else if (k->item_ < pattern) {
231  if (!k->right_) {
232  if (below) *below = false;
233  break;
234  }
235  j0 = j + 1;
236  k = k->right();
237  }
238  else { // item == k->data()
239  if (found) *found = true;
240  break;
241  }
242  }
243  }
244  if (index) *index = j;
245  return k2;
246 }
