62

64  if (protocol->downcase() == "ipv6") family = AF_INET6;
65  else if (protocol->downcase() == "ipv4") family = AF_INET;
66 
67  address_ = SocketAddressList::create();
68  if (address != "" && address != "*") {
69  Ref<SocketAddressList> l = SocketAddress::resolve(address, "http", family, SOCK_STREAM);
70 
71  for (int i = 0; i < l->count(); ++i) {
72  for (int j = 0; j < ports->count(); ++j) {
73  Ref<SocketAddress> a = SocketAddress::copy(l->at(i));
74  a->setPort(ports->at(j));
75  address_->append(a);
76  }
77  }
78  }
79  else {
80  if (family == AF_UNSPEC) {
81  for (int j = 0; j < ports->count(); ++j) {
82  address_->append(SocketAddress::create(AF_INET, "*", ports->at(j)));
83  address_->append(SocketAddress::create(AF_INET6, "*", ports->at(j)));
84  }
85  }
86  else {
87  for (int j = 0; j < ports->count(); ++j) {
88  address_->append(SocketAddress::create(family, "*", ports->at(j)));
89  }
90  }
91  }

97  SocketAddress *a = l->at(i);
