121 String Dir::createUnique(String path, int mode, char placeHolder)
122 {
123  Ref<Random> random = Random::open(Process::currentId());
124  while (true) {
125  String candidate = path->copy();
126  for (int i = 0, n = candidate->count(); i < n; ++i) {
127  if (candidate->at(i) == placeHolder) {
128  char r = random->get(0, 61);
129  if ((0 <= r) && (r <= 9))
130  r += '0';
131  else if ((10 <= r) && (r <= 35))
132  r += 'a' - 10;
133  else if ((36 <= r) && (r <= 61))
134  r += 'A' - 36;
135  candidate->at(i) = r;
136  }
137  }
138  if (::mkdir(candidate, mode) == -1) {
139  if (errno != EEXIST)
140  CC_SYSTEM_RESOURCE_ERROR(errno, candidate);
141  }
142  else {
143  return candidate;
144  }
145  }
146 }
