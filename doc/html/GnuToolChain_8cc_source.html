113 bool GnuToolChain::link(BuildPlan *plan)
114 {
115  String name = plan->name();
116  Version version = plan->version();
117  int options = plan->options();
118  ModuleList *modules = plan->modules();
119 
120  Format args;
121 
122  args << compiler();
123  if (plan->linkStatic()) args << "-static";
124  if ((options & BuildPlan::Library) && !plan->linkStatic()) args << "-shared";
125  args << "-pthread";
126 
127  if (options & BuildPlan::Library) {
128  args << (
129  Format() << "-Wl,-soname,lib" << name << ".so." << Version::major(version)
130  )->join();
131  }
132 
133  args << "-o" << linkName(plan);
134 
135  for (int i = 0; i < modules->count(); ++i)
136  args << modules->at(i)->modulePath();
137 
138  appendLinkOptions(args, plan);
139 
140  String command = args->join(" ");
141 
142  if (!plan->shell()->run(command))
143  return false;
144 
145  if ((options & BuildPlan::Library) && !plan->linkStatic())
146  createLibrarySymlinks(plan, linkName(plan));
147 
148  if (options & BuildPlan::Application)
149  createAliasSymlinks(plan, linkName(plan));
150 
151  return true;
152 }

154 bool GnuToolChain::testInclude(BuildPlan *plan, StringList *headers) const
155 {
156  String srcPath = File::createUnique("/tmp/XXXXXXXX.cc");
157  UnlinkGuard srcUnlinkGuard(srcPath);
158  UnlinkGuard objUnlinkGuard(srcPath->baseName() + ".o");
159  {
160  Ref<File> src = File::open(srcPath, File::WriteOnly);
161  Format format;
162  for (int i = 0; i < headers->count(); ++i)
163  format << "#include <" << headers->at(i) << ">\n";
164  format << "int main() { return 0; }\n";
165  src->write(format->join());
166  }
167 
168  Format args;
169  args << compiler();
170  appendCompileOptions(args, plan);
171  args << "-E" << srcPath;
172  String command = args->join(" ");
173  if (plan->options() & BuildPlan::Verbose)
174  fout() << "# " << command << nl;
175 
176  Ref<SubProcess> sub;
177  {
178  Ref<SubProcess::Params> params = SubProcess::Params::create();
179  params->setCommand(command);
180  Ref<SubProcess::Overloads> overloads = SubProcess::Overloads::create();
181  {
182  Ref<File> devNull = File::open("/dev/null", File::WriteOnly);
183  overloads->insert(StdOutFd, devNull);
184  if (!(plan->options() & BuildPlan::Verbose))
185  overloads->insert(StdErrFd, devNull);
186  }
187  sub = SubProcess::open(params);
188  }
189 
190  return sub->wait() == 0;
191 }

193 bool GnuToolChain::install(BuildPlan *plan)
194 {
195  int options = plan->options();
196  String product = linkName(plan);
197  String installDirPath = plan->installPath((options & BuildPlan::Library) ? "lib" : "bin");
198  String installFilePath = installDirPath->expandPath(product);
199 
200  if (!plan->shell()->install(product, installFilePath)) return false;
201 
202  if ((options & BuildPlan::Library) && !plan->linkStatic()) {
203  CwdGuard guard(installDirPath, plan->shell());
204  createLibrarySymlinks(plan, product);
205  }
206 
207  if (options & BuildPlan::Application && plan->alias()->count() > 0) {
208  CwdGuard guard(installDirPath, plan->shell());
209  createAliasSymlinks(plan, product);
210  }
211 
212  if (plan->bundle()->count() > 0) {
213  for (int i = 0; i < plan->bundle()->count(); ++i) {
214  String path = plan->bundle()->at(i);
215  String relativePath = path->copy(plan->projectPath()->count(), path->count());
216  plan->shell()->install(path, plan->installRoot()->expandPath(bundlePrefix(plan)->expandPath(relativePath)));
217  }
218  }
219 
220  return true;
221 }

289 void GnuToolChain::appendCompileOptions(Format args, BuildPlan *plan)
290 {
291  if (plan->options() & BuildPlan::Debug) args << "-g";
292  if (plan->options() & BuildPlan::Release) args << "-DNDEBUG";
293  if (plan->optimize() != "") args << "-O" + plan->optimize();
294  if (plan->linkStatic()) args << "-static";
295  if (!(
296  plan->customCompileFlags()->contains("-fPIC") ||
297  plan->customCompileFlags()->contains("-fPIE") ||
298  plan->customCompileFlags()->contains("-fpie") ||
299  plan->customCompileFlags()->contains("-fpic")
300  )) {
301  /*if (plan->options() & BuildPlan::Library) args << "-fPIC";
302  else*/ args << "-fPIC";
303  }
304  args << "-Wall" << "-pthread" << "-pipe";
305  if (plan->bundle()->count() > 0)
306  args << "-DCCBUILD_BUNDLE_PREFIX=" + bundlePrefix(plan);
307  if (plan->name() != "")
308  args << "-DCCBUILD_BUNDLE_VERSION=" + plan->version();
309  if (plan->customCompileFlags()) {
310  for (int i = 0; i < plan->customCompileFlags()->count(); ++i)
311  args << plan->customCompileFlags()->at(i);
312  }
313  for (int i = 0; i < plan->includePaths()->count(); ++i)
314  args << "-I" + plan->includePaths()->at(i);
315 }

317 void GnuToolChain::appendLinkOptions(Format args, BuildPlan *plan)
318 {
319  if (plan->customLinkFlags()) {
320  for (int i = 0; i < plan->customLinkFlags()->count(); ++i)
321  args << plan->customLinkFlags()->at(i);
322  }
323 
324  args << "-Wl,--no-as-needed";
325 
326  StringList *libraryPaths = plan->libraryPaths();
327  StringList *libraries = plan->libraries();
328 
329  if (!libraryPaths->contains("."))
330  args << "-L.";
331 
332  for (int i = 0; i < libraryPaths->count(); ++i)
333  args << "-L" + libraryPaths->at(i);
334 
335  for (int i = 0; i < libraries->count(); ++i)
336  args << "-l" + libraries->at(i);
337 
338  if (plan->containsCPlusPlus()) args << "-lstdc++";
339 
340  Ref<StringList> rpaths = StringList::create();
341  if (rpathOverride_ != "")
342  rpaths << "-rpath=" + rpathOverride_;
343  else
344  rpaths << "-rpath=$ORIGIN" << "-rpath=$ORIGIN/../lib";
345 
346  if (libraryPaths->count() > 0) {
347  for (int i = 0; i < libraryPaths->count(); ++i)
348  rpaths << "-rpath=" + libraryPaths->at(i)->absolutePathRelativeTo(Process::cwd());
349  }
350 
351  rpaths << "-rpath-link=" + Process::cwd();
352  args << "-Wl,--enable-new-dtags," + rpaths->join(",");
353 }
