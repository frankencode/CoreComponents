28 void YasonWriter::writeValue(Variant value, int depth)
29 {
30  if ( type(value) == Variant::IntType ||
31  type(value) == Variant::BoolType ||
32  type(value) == Variant::FloatType ||
33  type(value) == Variant::ColorType ||
34  type(value) == Variant::VersionType )
35  {
36  format_ << value;
37  }
38  else if (type(value) == Variant::StringType) {
39  String s = value;
40  if (s->contains("\""))
41  s = s->replace("\"", "\\\"");
42  s = s->escape();
43  format_ << "\"" << s << "\"";
44  }
45  else if (type(value) == Variant::ListType) {
46  writeList(value, depth);
47  }
48  else if (type(value) == Variant::ObjectType) {
49  writeObject(value, depth);
50  }
51 }

53 void YasonWriter::writeList(Variant value, int depth)
54 {
55  if (itemType(value) == Variant::IntType)
56  writeTypedList<int>(value, depth);
57  else if (itemType(value) == Variant::BoolType)
58  writeTypedList<bool>(value, depth);
59  else if (itemType(value) == Variant::FloatType)
60  writeTypedList<float>(value, depth);
61  else if (itemType(value) == Variant::StringType)
62  writeTypedList<String>(value, depth);
63  else
64  writeTypedList<Variant>(value, depth);
65 }

67 bool YasonWriter::isIdentifier(String name) const
68 {
69  for (int i = 0, n = name->count(); i < n; ++i) {
70  char ch = name->at(i);
71  if (!(
72  (('a' <= ch) && (ch <= 'z')) ||
73  (('A' <= ch) && (ch <= 'Z')) ||
74  (ch == '_') ||
75  (ch == '-') ||
76  (('0' <= ch) && (ch <= '9'))
77  ))
78  return false;
79  }
80  return true;
81 }

83 void YasonWriter::writeObject(Variant value, int depth)
84 {
85  Ref<MetaObject> object = cast<MetaObject>(value);
86  if (!object) {
87  format_ << "null";
88  return;
89  }
90  if (object->className() != "") {
91  format_ << object->className();
92  format_ << " ";
93  }
94  if (object->count() == 0 && !object->hasChildren()) {
95  format_ << "{}";
96  return;
97  }
98  format_ << "{\n";
99  writeIndent(depth + 1); // FIXME: having an "IndentStream" would be nice
100  for (int i = 0; i < object->count(); ++i) {
101  String memberName = object->keyAt(i);
102  Variant memberValue = object->valueAt(i);
103  if (isIdentifier(memberName))
104  format_ << memberName << ": ";
105  else
106  format_ << "\"" << memberName << "\": ";
107  writeValue(memberValue, depth + 1);
108  format_ << "\n";
109  if (i < object->count() - 1)
110  writeIndent(depth + 1);
111  }
112  if (object->hasChildren()) {
113  if (object->count() > 0) writeIndent(depth + 1);
114  MetaObjectList *children = object->children();
115  for (int i = 0; i < children->count(); ++i) {
116  writeObject(children->at(i), depth + 1);
117  format_ << "\n";
118  if (i < children->count() -1)
119  writeIndent(depth + 1);
120  }
121  }
122  writeIndent(depth);
123  format_ << "}";
124 }

133 {
134  List<T> *list = cast< List<T> >(value);
135  if (list->count() == 0) {
136  format_ << "[]";
137  return;
138  }
139  format_ << "[ ";

141  writeValue(list->at(i), depth);
142  if (i < list->count() - 1)
143  format_ << ", ";
144  }
145  format_ << " ]";
