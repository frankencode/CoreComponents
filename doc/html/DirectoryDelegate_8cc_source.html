32 void DirectoryDelegate::process(HttpRequest *request)
33 {
34  String path = directoryInstance_->path() + "/" + request->uri();
35  path = path->canonicalPath();
36  String prefix = path->head(directoryInstance_->path()->count());
37  if (path->head(directoryInstance_->path()->count()) != directoryInstance_->path()) throw Forbidden();
38 
39  if ((!directoryInstance_->showHidden()) && path->baseName()->beginsWith('.')) throw NotFound();
40 
41  Ref<FileStatus> fileStatus = FileStatus::read(path);
42 
43  if (scriptHandler_) {
44  if (
45  (!fileStatus->exists()) ||
46  ((fileStatus->mode() & AnyExec) && (fileStatus->type() != DirectoryType))
47  ) {
48  if (scriptHandler_->process(request, fileStatus, directoryInstance_->path()))
49  return;
50  }
51  }
52 
53  if (!fileStatus->exists()) throw NotFound();
54 
55  {
56  String h;
57  if (request->lookup("If-Modified-Since", &h)) {
58  Ref<Date> cacheDate = scanDate(h);
59  if (cacheDate) {
60  if (fileStatus->lastModified() <= cacheDate->time()) {
61  response()->setStatus(304);
62  response()->beginTransmission();
63  response()->endTransmission();
64  return;
65  }
66  }
67  }
68  }
69 
70  response()->setHeader("Last-Modified", formatDate(Date::create(fileStatus->lastModified())));
71 
72  if (fileStatus->type() == DirectoryType) {
73  String indexPath, indexName;
74  const char *candidateNames[] = { "index.html", "index.htm" };
75  for (int i = 0, n = sizeof(candidateNames) / sizeof(candidateNames[0]); i < n; ++i) {
76  String candidateName = candidateNames[i];
77  String candidatePath = path->expandPath(candidateName);
78  if (File::exists(candidatePath)) {
79  indexPath = candidatePath;
80  indexName = candidateName;
81  break;
82  }
83  }
84  if (indexPath != "") {
85  if (!request->uri()->endsWith('/')) {
86  // NOTE: not all user agents (e.g. FireFox) support Content-Location correctly
87  // therefore a redirection is the only clean way to make the user agent us a proper full path
88  // in subsequent requests
89  response()->setStatus(301);
90  response()->setHeader("Location", request->uri() + "/");
91  response()->beginTransmission();
92  response()->write("Moved permanently.");
93  response()->endTransmission();
94  return;
95  }
96 
97  deliverFile(indexPath);
98  }
99  else listDirectory(request, path);
100  }
101  else if (fileStatus->type() == RegularFileType && fileStatus->size() < 0x10000) {
102  deliverFile(path);
103  }
104  else {
105  streamFile(path);
106  }
107 }

109 void DirectoryDelegate::listDirectory(HttpRequest *request, String path)
110 {
111  Ref<Dir> dir = Dir::open(path);
112 
113  response()->setHeader("Content-Type", "text/html");
114  response()->chunk() <<
115  "<!DOCTYPE html>\n"
116  "<html>\n"
117  "<head>\n"
118  "<title>" << request->uri() << "</title>\n" <<
119  "<style type=\"text/css\">\n"
120  "a.file { float: left; width: 100pt; font-size: 10pt; padding: 10pt; overflow: hidden; text-overflow: ellipsis }\n"
121  "</style>\n"
122  "</head>\n"
123  "<body>\n";
124 
125  String prefix = request->uri()->canonicalPath();
126  if (prefix->count() > 0) {
127  if (prefix->at(prefix->count() - 1) != '/')
128  prefix += "/";
129  }
130 
131  Ref<StringList> entries = StringList::create();
132  for (String name; dir->read(&name);) {
133  if (name == "." || name == "..") continue;
134  if ((!directoryInstance_->showHidden()) && name->beginsWith('.')) continue;
135  entries->append(name);
136  }
137  entries = entries->sort();
138 
139  for (int i = 0; i < entries->count(); ++i) {
140  String name = entries->at(i);
141  response()->chunk() << "<a class=\"file\" href=\"" << prefix << name << "\">" << name << "</a>\n";
142  }
143 
144  response()->chunk() <<
145  "</body>\n"
146  "</html>\n";
147 }

159 void DirectoryDelegate::streamFile(String path)
160 {
161  Ref<File> file = File::open(path);
162  String head;
163  ssize_t size = -1;
164  if (file->seekable()) {
165  head = file->readSpan(64);
166  size = file->seek(0, File::SeekEnd);
167  file->seek(0, File::SeekBegin);
168  }
169  String mediaType = mediaTypeDatabase()->lookup(path, head);
170  if (mediaType != "") response()->setHeader("Content-Type", mediaType);
171  response()->beginTransmission(size);
172  Ref<ByteArray> buf = ByteArray::allocate(0x10000);
173  while (true) {
174  int n = file->read(buf);
175  if (n == 0) break;
176  response()->write(buf->select(0, n));
177  }
178  response()->endTransmission();
179 }
