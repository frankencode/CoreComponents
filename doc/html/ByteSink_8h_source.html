180 inline void ByteSink::writeUInt64(uint64_t x)
181 {
182  if (endian_ == LittleEndian) {
183  for (int i = 0; i < int(sizeof(x)); ++i, x >>= 8)
184  writeUInt8(uint8_t(x));
185  }
186  else {
187  for (int i = (int(sizeof(x))-1) * 8; i >= 0; i -= 8)
188  writeUInt8(uint8_t(x >> i));
189  }
