51  /** Insert a key-value mapping if no key-value mapping with the same key exists already.
52  * If currentValue is non-null the current value the giving key maps to is returned.
53  * The function returns true if the new key-value mapping was inserted successfully.
54  */
55  inline bool insert(const Key &key, const Value &value, Value *currentValue = 0, int *index = 0)
56  {
57  bool found = false;
58  bool below = true;
59  Item e(key, value);
60  Node *k = tree_.find(e, &found, &below, index);
61  if (found) {
62  if (currentValue)
63  *currentValue = k->item_.value();
64  }
65  else {
66  tree_.attach(k, new Node(e), below);
67  }
68  return !found;
69  }

95  /** Lookup key-value pair by given key.
96  * If a matching key-value pair is found the value is returned in 'value' and the
