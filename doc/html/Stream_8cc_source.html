34 off_t Stream::transferSpanTo(off_t count, Stream *sink, ByteArray *buffer)
35 {
36  if (count == 0) return 0;
37 
38  off_t total = 0;
39  Ref<ByteArray> h;
40  if (!buffer) {
41  h = ByteArray::allocate((0 < count && count < 0x4000) ? count : 0x4000);
42  buffer = h;
43  }
44 
45  while (true) {
46  int n = read(buffer);
47  if (n == 0) break;
48  if (sink) sink->write(buffer->select(0, n));
49  total += n;
50  if (count > 0) {
51  count -= n;
52  if (count == 0) break;
53  if (count < buffer->count()) buffer->truncate(count);
54  }
55  }
56 
57  return total;
58 }
