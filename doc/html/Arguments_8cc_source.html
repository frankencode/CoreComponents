21  options_(VariantMap::create()),

51 void Arguments::validate(VariantMap *prototype)
52 {
53  for (int i = 0; i < options_->count(); ++i)
54  {
55  String name = options_->keyAt(i);
56  Variant value = options_->valueAt(i);
57 
58  Variant defaultValue;
59  if (!prototype->lookup(name, &defaultValue))
60  throw UsageError(Format("No such option: \"%%\"") << name);
61  if (defaultValue == Variant()) continue;
62  int valueType = type(value);
63  int defaultType = type(defaultValue);
64  if (valueType != defaultType) {
65  if (valueType == Variant::IntType && defaultType == Variant::BoolType) {
66  int intValue = value;
67  if (intValue == 0 || intValue == 1)
68  options_->establish(name, intValue == 1);
69  }
70  else if (valueType == Variant::IntType && defaultType == Variant::FloatType) {
71  options_->establish(name, float(int(value)));
72  }
73  else if (valueType != Variant::ObjectType && defaultType == Variant::StringType) {
74  options_->establish(name, str(value));
75  }
76  else {
77  throw UsageError(
78  Format("Option \"%%\" expects type %%") << name << Variant::typeName(type(defaultValue), itemType(defaultValue))
79  );
80  }
81  }
82  }
83 }
