68 void HttpSocket::sessionInit()
69 {
70  CC_ASSERT(mode_ & Connected);
71 
72  t0_ = System::now();
73  te_ = t0_ + nodeConfig()->connectionTimeout();
74 
75  if (mode_ & Plaintext) return;
76 
77  CC_ASSERT(!(mode_ & Open));
78 
79  gnuTlsCheckSuccess(gnutls_init(&session_, GNUTLS_SERVER));
80  if (nodeConfig()->security()->hasCredentials())
81  gnuTlsCheckSuccess(gnutls_credentials_set(session_, GNUTLS_CRD_CERTIFICATE, nodeConfig()->security()->cred_));
82  if (nodeConfig()->security()->hasCiphers())
83  gnuTlsCheckSuccess(gnutls_priority_set(session_, nodeConfig()->security()->prio_));
84  else
85  gnuTlsCheckSuccess(gnutls_set_default_priority(session_));
86  gnutls_handshake_set_post_client_hello_function(session_, onClientHello);
87  gnutls_transport_set_ptr(session_, this);
88  gnutls_transport_set_pull_function(session_, gnuTlsPull);
89  gnutls_transport_set_vec_push_function(session_, gnuTlsPushVec);
90 
91  SecurityMaster::instance()->prepareSessionResumption(session_);
92 }

99  void selectService(gnutls_session_t session)
100  {
101  size_t size = 0;
102  unsigned type = 0;
103  int ret = gnutls_server_name_get(session, NULL, &size, &type, 0);
104  if (type == GNUTLS_NAME_DNS) {
105  CC_ASSERT(ret == GNUTLS_E_SHORT_MEMORY_BUFFER);
106  serverName_ = String(size);
107  ret = gnutls_server_name_get(session, serverName_->bytes(), &size, &type, 0);
108  if (ret != GNUTLS_E_SUCCESS) {
109  CCNODE_ERROR() << peerAddress_ << ": " << gnutls_strerror(ret) << nl;
110  serverName_ = "";
111  }
112  if (serverName_->count() > 0) {
113  if (serverName_->at(serverName_->count() - 1) == 0)
114  serverName_->truncate(serverName_->count() - 1);
115  if (errorLog()->infoStream() != NullStream::instance())
116  CCNODE_INFO() << "TLS client hello: SNI=\"" << serverName_ << "\"" << nl;
117  serviceInstance_ = nodeConfig()->selectService(serverName_);
118  if (serviceInstance_) {
119  if (serviceInstance_->security()->hasCredentials())
120  HttpSocket::gnuTlsCheckSuccess(gnutls_credentials_set(session, GNUTLS_CRD_CERTIFICATE, serviceInstance_->security()->cred_), peerAddress_);
121  if (serviceInstance_->security()->hasCiphers())
122  HttpSocket::gnuTlsCheckSuccess(gnutls_priority_set(session, serviceInstance_->security()->prio_), peerAddress_);
123  }
124  }
125  }
126  }

154 ServiceInstance *HttpSocket::handshake()
155 {
156  CC_ASSERT(mode_ & Connected);
157 
158  if (mode_ & Plaintext) return 0;
159 
160  CC_ASSERT(!(mode_ & Open));
161 
162  ClientHelloContext::instance()->reset(address());
163 
164  while (true) {
165  int ret = gnutls_handshake(session_);
166  if (ret == GNUTLS_E_SUCCESS) break;
167  if (ret < 0) {
168  if (gnutls_error_is_fatal(ret))
169  gnuTlsCheckSuccess(ret);
170  }
171  }
172 
173  mode_ |= Open;
174 
175  if (errorLog()->infoStream() != NullStream::instance()) {
176  double t1 = System::now();
177  CCNODE_INFO() << "TLS handshake took " << int(1000 * (t1 - t0_)) << "ms" << nl;
178  }
179 
180  ServiceInstance *serviceInstance = ClientHelloContext::instance()->serviceInstance();
181  if (!serviceInstance) {
182  if (nodeConfig()->serviceInstances()->count() == 1)
183  serviceInstance = nodeConfig()->serviceInstances()->at(0);
184  if (ClientHelloContext::instance()->serverName() == "")
185  serviceInstance = nodeConfig()->selectService("");
186  }
187 
188  return serviceInstance;
189 }

217 void HttpSocket::write(const ByteArray *data)
218 {
219  if (data->count() == 0) return;
220 
221  if (mode_ & Plaintext) {
222  StreamSocket::write(data);
223  return;
224  }
225 
226  String pending = data;
227  while (true) {
228  int ret = gnutls_record_send(session_, pending->bytes(), pending->count());
229  if (ret == pending->count()) break;
230  gnuTlsCheckError(ret);
231  CC_ASSERT(ret > 0);
232  if (ret < pending->count())
233  pending = pending->select(ret, pending->count());
234  }
235 }
