199 Ref<Matches> findMatches(ByteArray *text, SyntaxDefinition *textPattern)
200 {
201  Ref<Matches> matches = Matches::create();
202  int ln = 1;
203  for (int i = 0; i < text->count();) {
204  Ref<SyntaxState> state = textPattern->find(text, i);
205  if (!state->valid()) break;
206  int i0 = state->i0();
207  int i1 = state->i1();
208  Ref<Range> capture;
209  if (state->lookupCapture("", &capture)) {
210  i0 = capture->i0();
211  i1 = capture->i1();
212  }
213  for (;i < i0; ++i)
214  if (text->at(i) == '\n') ++ln;
215  matches->append(Match::create(ln, i0, i1));
216  for (;i < i1; ++i)
217  if (text->at(i) == '\n') ++ln;
218  if (i0 == i1) ++i;
219  }
220  return matches;
221 }

223 void displayMatch(ByteArray *path, ByteArray *text, Match *match)
224 {
225  int ln = match->ln();
226  int i0 = match->i0();
227  int i1 = match->i1();
228 
229  int j0 = i0;
230  for (;j0 > 0; --j0)
231  if (text->at(j0 - 1) == '\n') break;
232 
233  fout("%%:") << path;
234  if (i0 == i1) {
235  fout("%%: %%\n") << ln << text->copy(j0, text->find("\n", i0));
236  return;
237  }
238 
239  bool multiline = text->copy(i0, i1)->contains('\n');
240  if (multiline) fout() << nl;
241 
242  for (int j1 = j0; j0 < i1; j0 = j1) {
243  for (;j1 < text->count(); ++j1)
244  if (text->at(j1) == '\n') break;
245  Format line;
246  line << ln << ": ";
247  int k0 = j0, k1 = j1;
248  if (j0 <= i0 && i0 < j1) k0 = i0;
249  if (j0 < i1 && i1 < j1) k1 = i1;
250  if (j0 < k0) line << text->copy(j0, k0);
251  if (k0 < k1) line << "\033[7m" << text->copy(k0, k1) << "\033[m";
252  if (k1 < j1) line << text->copy(k1, j1);
253  line << "\n";
254  fout() << line->join();
255  ++ln;
256  ++j1;
257  }

260 String replaceMatches(ByteArray *text, Matches *matches, ByteArray *replacement)
261 {
262  Ref<StringList> fragments = StringList::create();
263  int fi0 = 0; // begin of fragment
264  int si = 0, sl = 0; // index and line shift
265  int nr = replacement->count('\n');
266  for (int i = 0; i < matches->count(); ++i) {
267  Match *match = matches->at(i);
268  fragments->append(text->copy(fi0, match->i0()));
269  fi0 = match->i1();
270  int i0s = match->i0() + si;
271  si += replacement->count() - (match->i1() - match->i0());
272  sl += nr - text->copy(match->i0(), match->i1())->count('\n');
273  match->moveTo(match->ln() + sl, i0s, i0s + replacement->count());
274  }
275  fragments->append(text->copy(fi0, text->count()));
276  return fragments->join(replacement);
