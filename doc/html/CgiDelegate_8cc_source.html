55 void CgiDelegate::process(HttpRequest *request, String script, String documentRoot)
56 {
57  Ref<CgiServerConnection> cgiServer;
58  Ref<SubProcess> sub;
59 
60  String payload;
61  {
62  int payloadSize = 0; {
63  String h;
64  if (request->lookup("Content-Length", &h))
65  payloadSize = h->toNumber<int>();
66  }
67 
68  if (payloadSize > 0) {
69  if (payloadSize > cgiInstance_->requestPayloadLimit())
70  throw PayloadTooLarge();
71  payload = request->payload()->readSpan(payloadSize);
72  }
73  }
74 
75  if (cgiInstance_->serverAddress()->count() > 0)
76  {
77  int nextIndex = (nextPeer_++) % cgiInstance_->serverAddress()->count();
78  Ref<SocketAddress> address = cgiInstance_->serverAddress()->at(nextIndex);
79  if (cgiInstance_->serverAddress()->count() > 1)
80  CCNODE_DEBUG() << "Forwarding request to server " << address << nl;
81 
82  cgiServer = CgiServerConnection::open(StreamSocket::connect(address));
83  if (errorLog()->level() >= DebugLogLevel)
84  cgiServer->setupTransferLog(address->toString());
85 
86  String headerText = compileHeader(request, payload);
87  cgiServer->stream()->write(
88  Format()
89  << headerText->count() << ":"
90  << headerText << "," << payload
91  );
92  }
93  else if (script != "")
94  {
95  Ref<StringList> args = script->split(' ');
96  String scriptPath = args->at(0);
97  if (!scriptPath->isAbsolutePath()) scriptPath = documentRoot->expandPath(scriptPath);
98  args->at(0) = scriptPath;
99 
100  Ref<EnvMap> env = makeEnv(request, payload);
101  env->insert("SCRIPT_NAME", "/" + scriptPath->baseName());
102  if (documentRoot != "") env->insert("DOCUMENT_ROOT", documentRoot);
103 
104  sub = SubProcess::open(
105  SubProcess::params()
106  ->setArgs(args)
107  ->setEnvMap(env)
108  );
109 
110  cgiServer = CgiServerConnection::open(sub);
111  if (errorLog()->level() >= DebugLogLevel)
112  cgiServer->setupTransferLog(scriptPath->baseName());
113 
114  if (payload->count() > 0)
115  cgiServer->stream()->write(payload);
116 
117  sub->shutdown(SubProcess::WriteShutdown);
118  }
119  else
120  throw InternalServerError();
121 
122  Ref<CgiResponse> cgiResponse = cgiServer->readResponse();
123 
124  int statusCode = -1;
125  String reasonPhrase;
126  {
127  String statusField;
128  int fieldIndex = 0;
129  if (cgiResponse->lookup("Status", &statusField, &fieldIndex)) {
130  int i = statusField->find(' ');
131  statusCode = statusField->select(0, i)->toNumber<int>();
132  if (i < statusField->count())
133  reasonPhrase = statusField->copy(i + 1, statusField->count());
134  if (statusCode == 0) {
135  CCNODE_ERROR() << "Mailformed Status field in CGI response: \"" << statusField << "\"" << nl;
136  throw InternalServerError();
137  }
138  cgiResponse->removeAt(fieldIndex);
139  }
140  }
141 
142  {
143  String location;
144  int fieldIndex = 0;
145  if (cgiResponse->lookup("Location", &location, &fieldIndex)) {
146  CCNODE_DEBUG() << "Redirect to \"" << location << "\"" << nl;
147  if (location->beginsWith("/")) {
148  cgiResponse->removeAt(fieldIndex);
149  String content = File::open(location)->map();
150  String contentType;
151  if (!cgiResponse->lookup("Content-Type", &contentType)) {
152  contentType = mediaTypeDatabase()->lookup(location, content);
153  cgiResponse->insert("Content-Type", contentType);
154  }
155  response()->setStatus(statusCode, reasonPhrase);
156  for (int i = 0; i < cgiResponse->count(); ++i)
157  response()->setHeader(cgiResponse->keyAt(i), cgiResponse->valueAt(i));
158  response()->beginTransmission(content->count());
159  response()->write(content);
160  response()->endTransmission();
161  return;
162  }
163  else if (statusCode < 0) {
164  statusCode = 302;
165  reasonPhrase = "Found";
166  }
167  }
168  }
169 
170  if (statusCode < 0) {
171  statusCode = 200;
172  reasonPhrase = "OK";
173  }
174 
175  response()->setStatus(statusCode, reasonPhrase);
176  ssize_t contentLength = -1;
177  {
178  String value;
179  int i = 0;
180  if (cgiResponse->lookup("Content-Length", &value, &i)) {
181  contentLength = value->toNumber<size_t>();
182  cgiResponse->remove("Content-Length");
183  }
184  }
185  response()->setHeader(cgiResponse);
186  autoSecureForwardings();
187 
188  if (request->majorVersion() == 1 && request->minorVersion() == 0)
189  {
190  Ref<ByteArray> content = cgiResponse->payload()->readSpan(contentLength);
191  response()->beginTransmission(content->count());
192  response()->write(content);
193  response()->endTransmission();
194  }
195  else {
196  response()->beginTransmission();
197  ssize_t totalTransferred = 0;
198  {
199  Ref<ByteArray> buffer = ByteArray::allocate(0x10000);
200  while (true) {
201  int n = cgiResponse->payload()->read(buffer);
202  if (n == 0) break;
203  response()->write(buffer->select(0, n));
204  totalTransferred += n;
205  if (contentLength >= 0) {
206  if (totalTransferred >= contentLength) break;
207  }
208  }
209  response()->endTransmission();
210  }
211  }
212 
213  if (sub) {
214  int ret = sub->wait();
215  if (ret != 0)
216  CCNODE_ERROR() << "CGI process return error, ret = " << ret << nl;
217  }
218 }

220 Ref<CgiDelegate::EnvMap> CgiDelegate::makeEnv(HttpRequest *request, ByteArray *payload) const
221 {
222  String queryString = urlDecode(request, payload);
223 
224  Ref<EnvMap> env = EnvMap::create();
225  env->insert("CONTENT_LENGTH", str(payload->count()));
226  env->insert("REQUEST_METHOD", request->method());
227  env->insert("REQUEST_URI", request->uri());
228  env->insert("QUERY_STRING", queryString);
229  env->insert("REMOTE_ADDR", client()->address()->networkAddress());
230  env->insert("REMOTE_PORT", str(client()->address()->port()));
231  env->insert("GATEWAY_INTERFACE", "CGI/1.1");
232  env->insert("SERVER_PROTOCOL", request->version());
233  env->insert("SERVER_NAME", request->host());
234  env->insert("SERVER_SOFTWARE", nodeConfig()->version());
235 
236  {
237  String h;
238  if (request->lookup("Content-Type", &h))
239  env->insert("CONTENT_TYPE", h);
240  }
241 
242  if (client()->isSecure()) {
243  env->insert("HTTP_X_FORWARDED_PROTO", "https");
244  env->insert("HTTPS", "on");
245  }
246 
247  for (int i = 0; i < request->count(); ++i)
248  env->insert(wrapHttp(request->keyAt(i)), request->valueAt(i));
249 
250  Ref<VariantMap> userEnv = cgiInstance_->environment();
251  for (int i = 0; i < userEnv->count(); ++i)
252  env->insert(userEnv->keyAt(i), userEnv->valueAt(i));
253 
254  return env;
255 }

274  << "REMOTE_ADDR" << client()->address()->networkAddress()
275  << "REMOTE_PORT" << client()->address()->port()
276  << "GATEWAY_INTERFACE" << "CGI/1.1"

278  << "SERVER_NAME" << request->host()

280 
281  {
282  String h;

284  header << "CONTENT_TYPE" << h;
285  }
286 
287  if (client()->isSecure()) {
288  header
289  << "HTTP_X_FORWARDED_PROTO" << "https"
290  << "HTTPS" << "on";
291  }
292

294  header << wrapHttp(request->keyAt(i)) << request->valueAt(i);
295 
296  Ref<VariantMap> userEnv = cgiInstance_->environment();
297  for (int i = 0; i < userEnv->count(); ++i)
298  header << userEnv->keyAt(i) << userEnv->valueAt(i);
299 
300  header << "";
301 
302  return ByteArray::join(header, '\0');

305 String CgiDelegate::urlDecode(HttpRequest *request, ByteArray *payload)
306 {
307  String queryString = request->query();
308  if (request->method() == "POST") {
309  if (request->value("Content-Type") == "application/x-www-form-urlencoded") {
310  queryString = payload->copy(); // FIXME: not strict enough
311  request->remove("Content-Type"); // FIXME: use removeAt or place contentType() into HttpRequest
312  request->establish("Content-Length", "0");
313  payload->truncate(0);
314  }
315  }
316  return queryString;
