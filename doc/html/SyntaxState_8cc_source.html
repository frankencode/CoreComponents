58 bool SyntaxState::lookupCapture(const char *name, Ref<Range> *range) const
59 {
60  const DefinitionNode *scope = definition_->resolveScope(name);
61  int captureId = -1;
62  if (!scope->lookupCaptureIdByName(name, &captureId)) return false;
63  CC_ASSERT(captureId != -1);
64  if (scope == definition_) {
65  *range = captures_->at(captureId);
66  return true;
67  }
68  Ref<SyntaxState> state;
69  if (!stateByScope_->lookup(scope, &state)) return false;
70  *range = state->captures_->at(captureId);
71  return true;
72 }

74 SyntaxState *SyntaxState::stateByScope(const DefinitionNode *scope)
75 {
76  if (scope == definition_) return this;
77  if (!stateByScope_) stateByScope_ = StateByScope::create();
78  Ref<SyntaxState> state;
79  if (!stateByScope_->lookup(scope, &state)) {
80  state = scope->createState(tokenFactory_);
81  stateByScope_->insert(scope, state);
82  }
83  return state;
